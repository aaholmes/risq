13/// Create a simple 4-electron, 4-orbital test system
14pub fn create_test_system() -> (Ham, VarWf, ExciteGenerator, Global) {
15    use crate::config::GlobalConfig;
16    use crate::ham::read_ints::read_ints;
17    use std::io::Write;
18    use std::fs::File;
19    
20    // Create a minimal FCIDUMP file for 4 electrons in 4 orbitals
21    let fcidump_content = "&FCI NORB=4,NELEC=4,MS2=0,\n  ORBSYM=1,1,1,1,\n  ISYM=1,\n&END\n0.500000000000000E+00   1   1   1   1\n0.200000000000000E+00   1   1   2   2\n0.200000000000000E+00   2   2   1   1\n0.200000000000000E+00   2   2   2   2\n0.100000000000000E+00   1   2   1   2\n0.100000000000000E+00   1   2   2   1\n0.100000000000000E+00   2   1   1   2\n0.100000000000000E+00   2   1   2   1\n0.200000000000000E+00   3   3   3   3\n0.100000000000000E+00   3   3   4   4\n0.100000000000000E+00   4   4   3   3\n0.200000000000000E+00   4   4   4   4\n0.050000000000000E+00   3   4   3   4\n0.050000000000000E+00   3   4   4   3\n0.050000000000000E+00   4   3   3   4\n0.050000000000000E+00   4   3   4   3\n-1.000000000000000E+00   1   1   0   0\n-0.800000000000000E+00   2   2   0   0\n-0.600000000000000E+00   3   3   0   0\n-0.400000000000000E+00   4   4   0   0\n0.100000000000000E+00   1   2   0   0\n0.100000000000000E+00   2   1   0   0\n0.050000000000000E+00   3   4   0   0\n0.050000000000000E+00   4   3   0   0\n0.000000000000000E+00   0   0   0   0\n";
22    
23    let temp_path = "/tmp/debug_fcidump.txt";
24    let mut file = File::create(temp_path).expect("Failed to create temp FCIDUMP file");
25    file.write_all(fcidump_content.as_bytes()).expect("Failed to write FCIDUMP");
26    
27    let config = GlobalConfig {
28        n_orbs: 4,
29        n_core: 0,
30        n_up: 2,
31        n_dn: 2,
32        n_states: 1,
33        z_sym: 1,
34    };
35    
36    let ham = read_ints(&config, temp_path).expect("Failed to read FCIDUMP");
37    
38    // Create a small set of determinants to test
39    let determinants = vec![
40        create_config(&[0, 1], &[0, 1]),  // HF reference: |01⟩|01⟩
41        create_config(&[0, 2], &[0, 1]),  // Single excitation: α(1→2)
42        create_config(&[0, 1], &[0, 2]),  // Single excitation: β(1→2)
43        create_config(&[0, 2], &[0, 2]),  // Double excitation: both α(1→2), β(1→2)
44        create_config(&[1, 2], &[0, 1]),  // Single excitation: α(0→2)
45        create_config(&[0, 1], &[1, 2]),  // Single excitation: β(0→2)
46    ];
47    
48    // Create wavefunction
49    let mut dets = Vec::new();
50    let mut inds = HashMap::new();
51    for (i, config) in determinants.iter().enumerate() {
52        // Compute diagonal element for this determinant
53        let diagonal = ham.ham_diag(config);
54        dets.push(crate::wf::det::Det {
55            config: *config,
56            coeff: if i == 0 { 1.0 } else { 0.0 },
57            diag: Some(diagonal),
58        });
59        inds.insert(*config, i);
60    }
61    
62    let wf = Wf {
63        n: determinants.len(),
64        dets,
65        inds,
66        energy: -1.0,
67    };
68    
69    let mut var_wf = VarWf::default();
70    var_wf.wf = wf;
71    var_wf.new_sparse_ham();
72    
73    // Create minimal global and excite_gen
74    let global = Global {
75        norb: 4,
76        norb_core: 0,
77        nup: 2,
78        ndn: 2,
79        z_sym: 1,
80        n_states: 1,
81        eps_var: 1e-4,
82        eps_pt_dtm: 1e-6,
83        opp_algo: 1,
84        same_algo: 1,
85        target_uncertainty: 1e-5,
86        n_samples_per_batch: 1000,
87        n_batches: 100,
88        n_cross_term_samples: 1000,
89        use_new_semistoch: true,
90    };
91    let excite_gen = ExciteGenerator::default();
92    
93    (ham, var_wf, excite_gen, global)
94}

96fn create_config(up_orbs: &[i32], dn_orbs: &[i32]) -> Config {
97    let mut up = 0u128;
98    let mut dn = 0u128;
99    
100    for &orb in up_orbs {
101        up = ibset(up, orb);
102    }
103    for &orb in dn_orbs {
104        dn = ibset(dn, orb);
105    }
106    
107    Config { up, dn }
108}

110/// Compare Hamiltonian matrices from both algorithms
111pub fn debug_hamiltonian_differences() {
112    println!("=== Debugging Fast vs Original Hamiltonian Construction ===");
113    
114    let (ham, mut var_wf, excite_gen, global) = create_test_system();
115    
116    println!("Test system: {} determinants", var_wf.wf.n);
117    for (i, det) in var_wf.wf.dets.iter().enumerate() {
118        println!("  Det {}: α={:04b} β={:04b}", i, det.config.up as u8, det.config.dn as u8);
119    }
120    
121    // Create two separate wavefunction instances
122    let (_, mut wf_original, _, _) = create_test_system();
123    let (_, mut wf_fast, _, _) = create_test_system();
124    
125    // Run original algorithm
126    println!("\n--- Running Original Algorithm ---");
127    gen_sparse_ham_fast(&global, &mut wf_original, &ham, &excite_gen);
128    
129    let original_elements = extract_matrix_elements(&wf_original);
130    println!("Original algorithm found {} off-diagonal elements", original_elements.len());
131    
132    // Run fast algorithm
133    println!("\n--- Running Fast Algorithm ---");
134    gen_sparse_ham_fast_lookup(&global, &mut wf_fast, &ham);
135    
136    let fast_elements = extract_matrix_elements(&wf_fast);
137    println!("Fast algorithm found {} off-diagonal elements", fast_elements.len());
138    
139    // Compare results
140    println!("\n--- Comparing Results ---");
141    compare_matrix_elements(&original_elements, &fast_elements, &wf_original);
142    
143    // Print full matrices for detailed analysis
144    println!("\n--- Original Algorithm Matrix Elements ---");
145    print_matrix_elements(&original_elements, &wf_original);
146    
147    println!("\n--- Fast Algorithm Matrix Elements ---");
148    print_matrix_elements(&fast_elements, &wf_fast);
149}

151fn extract_matrix_elements(wf: &VarWf) -> Vec<(usize, usize, f64)> {
152    let mut elements = Vec::new();
153    
154    for i in 0..wf.sparse_ham.n {
155        for &(j, val) in &wf.sparse_ham.off_diag[i] {
156            elements.push((i, j, val));
157        }
158    }
159    
160    elements.sort_by(|a, b| a.0.cmp(&b.0).then(a.1.cmp(&b.1)));
161    elements
162}

164fn compare_matrix_elements(
165    original: &[(usize, usize, f64)], 
166    fast: &[(usize, usize, f64)],
167    wf: &VarWf
168) {
169    let original_set: HashSet<(usize, usize)> = original.iter().map(|(i, j, _)| (*i, *j)).collect();
170    let fast_set: HashSet<(usize, usize)> = fast.iter().map(|(i, j, _)| (*i, *j)).collect();
171    
172    let missing_in_fast: Vec<_> = original_set.difference(&fast_set).collect();
173    let extra_in_fast: Vec<_> = fast_set.difference(&original_set).collect();
174    
175    println!("Missing from fast algorithm: {} elements", missing_in_fast.len());
176    for &(i, j) in missing_in_fast.iter().take(10) {
177        let det_i = &wf.wf.dets[*i].config;
178        let det_j = &wf.wf.dets[*j].config;
179        let original_val = original.iter().find(|(ii, jj, _)| ii == i && jj == j).map(|(_, _, v)| *v).unwrap_or(0.0);
180        println!("  ({}, {}): {:04b}|{:04b} <-> {:04b}|{:04b} = {:.6}", 
181            i, j, 
182            det_i.up as u8, det_i.dn as u8, 
183            det_j.up as u8, det_j.dn as u8,
184            original_val
185        );
186        
187        // Analyze excitation type
188        let excitation_type = analyze_excitation_type(det_i, det_j);
189        println!("      Excitation type: {}", excitation_type);
190    }
191    
192    println!("Extra in fast algorithm: {} elements", extra_in_fast.len());
193    for &(i, j) in extra_in_fast.iter().take(10) {
194        let det_i = &wf.wf.dets[*i].config;
195        let det_j = &wf.wf.dets[*j].config;
196        let fast_val = fast.iter().find(|(ii, jj, _)| ii == i && jj == j).map(|(_, _, v)| *v).unwrap_or(0.0);
197        println!("  ({}, {}): {:04b}|{:04b} <-> {:04b}|{:04b} = {:.6}", 
198            i, j, 
199            det_i.up as u8, det_i.dn as u8, 
200            det_j.up as u8, det_j.dn as u8,
201            fast_val
202        );
203        
204        let excitation_type = analyze_excitation_type(det_i, det_j);
205        println!("      Excitation type: {}", excitation_type);
206    }
207    
208    // Check for value differences in common elements
209    let mut value_differences = 0;
210    for (i, j, original_val) in original {
211        if let Some((_, _, fast_val)) = fast.iter().find(|(ii, jj, _)| ii == i && jj == j) {
212            if (original_val - fast_val).abs() > 1e-12 {
213                value_differences += 1;
214                println!("Value difference at ({}, {}): original={:.10}, fast={:.10}", 
215                    i, j, original_val, fast_val);
216            }
217        }
218    }
219    
220    if value_differences == 0 {
221        println!("✅ All common matrix elements have identical values");
222    } else {
223        println!("❌ {} matrix elements have different values", value_differences);
224    }
225}

227fn analyze_excitation_type(det1: &Config, det2: &Config) -> String {
228    let up_diff = (det1.up ^ det2.up).count_ones();
229    let dn_diff = (det1.dn ^ det2.dn).count_ones();
230    
231    match (up_diff, dn_diff) {
232        (0, 0) => "identical".to_string(),
233        (2, 0) => "alpha single".to_string(),
234        (0, 2) => "beta single".to_string(),
235        (4, 0) => "alpha double".to_string(),
236        (0, 4) => "beta double".to_string(),
237        (2, 2) => "opposite-spin double".to_string(),
238        _ => format!("higher order ({} up, {} dn)", up_diff, dn_diff),
239    }
240}

242fn print_matrix_elements(elements: &[(usize, usize, f64)], wf: &VarWf) {
243    for (i, j, val) in elements {
244        let det_i = &wf.wf.dets[*i].config;
245        let det_j = &wf.wf.dets[*j].config;
246        let excitation_type = analyze_excitation_type(det_i, det_j);
247        println!("  H[{},{}] = {:8.6} | {:04b}|{:04b} <-> {:04b}|{:04b} | {}", 
248            i, j, val,
249            det_i.up as u8, det_i.dn as u8, 
250            det_j.up as u8, det_j.dn as u8,
251            excitation_type
252        );
253    }
