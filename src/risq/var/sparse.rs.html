1//! # Custom Sparse Matrix Representations (`var::sparse`)
2//!
3//! This module defines custom data structures for representing the sparse Hamiltonian matrix
4//! used in the variational HCI calculation. It provides implementations of the
5//! `MatrixOperations` trait required by the Davidson eigenvalue solver.
6//!
7//! Different structures might represent different storage strategies or optimizations.
8
9extern crate nalgebra;
10extern crate sprs;
11use nalgebra::base::{DMatrixSlice, DVector, DVectorSlice};
12use nalgebra::DMatrix;
13
14use crate::var::eigenvalues::matrix_operations::MatrixOperations;

16/// Represents a symmetric sparse matrix storing only the upper triangle.
17///
18/// This structure stores the diagonal elements explicitly and the off-diagonal
19/// elements as a vector of vectors (adjacency list format). `off_diag[i]` contains
20/// a list of tuples `(j, H_ij)` where `j > i`.
21/// This format allows for efficient addition of elements during Hamiltonian generation,
22/// as sorting is only needed once before diagonalization.
23#[derive(Default, Debug)] // Added Debug derive
24pub struct SparseMatUpperTri {
25    /// Dimension of the square matrix (N x N).
26    pub(crate) n: usize,
27    /// Vector storing the diagonal elements `H_ii`.
28    pub(crate) diag: Vec<f64>,
29    /// Tracks the number of non-zero off-diagonal elements currently stored *per row*
30    /// *before* deduplication. Used internally during construction.
31    pub(crate) nnz: Vec<usize>,
32    /// Stores the non-zero upper triangular off-diagonal elements.
33    /// `off_diag[i]` is a `Vec<(usize, f64)>` containing pairs `(j, H_ij)` where `j > i`.
34    /// Elements might be unsorted or contain duplicates during construction.
35    pub off_diag: Vec<Vec<(usize, f64)>>,
36}
37
38impl SparseMatUpperTri {
39    /// Sorts the off-diagonal elements for each row by column index and removes duplicates.
40    ///
41    /// This should be called after generating all Hamiltonian elements and before using
42    /// the matrix in operations like matrix-vector products (e.g., within Davidson).
43    /// Updates the internal `nnz` count after deduplication.
44    pub fn sort_remove_duplicates(&mut self) {
45        let mut n_upper_t = 0;
46        for (i, v) in self.off_diag.iter_mut().enumerate() {
47            // Sort by index
48            v[self.nnz[i]..].sort_by_key(|k| k.0);
49            // Call dedup on everything because I can't figure out how to do dedup on a slice
50            // (but should be fast enough anyway)
51            v.dedup_by_key(|k| k.0);
52            self.nnz[i] = v.len();
53            n_upper_t += self.nnz[i];
54        }
55        println!("Variational Hamiltonian has {} nonzero off-diagonal elements in upper triangle, {} total", n_upper_t, 2 * n_upper_t);
56    }
57}
58
59impl MatrixOperations for SparseMatUpperTri {
60    /// Computes the matrix-vector product `y = A*x` where `A` is `self` and `x` is `vs`.
61    /// Exploits the symmetry and upper triangular storage:
62    /// `y_i = A_ii*x_i + sum_{j>i} A_ij*x_j + sum_{j<i} A_ji*x_j`
63    /// where `A_ji = A_ij` is retrieved from `self.off_diag[j]`.
64    fn matrix_vector_prod(&self, vs: DVectorSlice<'_, f64>) -> DVector<f64> {
65        let mut res: DVector<f64> = DVector::from(vec![0.0; self.n]);
66
67        for row_ind in (0..self.n).into_iter() {
68            // Diagonal component
69            res[row_ind] += self.diag[row_ind] * vs[row_ind];
70
71            // Off-diagonal component
72            for (ind, val) in self.off_diag[row_ind].iter() {
73                res[row_ind] += val * vs[*ind];
74                res[*ind] += val * vs[row_ind];
75            }
76        }
77        res
78    }
79
80    /// Computes the matrix-matrix product `C = A*B` where `A` is `self` and `B` is `mtx`.
81    /// Performs the product column by column using `matrix_vector_prod`.
82    fn matrix_matrix_prod(&self, mtx: DMatrixSlice<'_, f64>) -> DMatrix<f64> {
83        let mut res: DMatrix<f64> = mtx.clone().into();
84        for (in_col, mut out_col) in mtx.column_iter().zip(res.column_iter_mut()) {
85            out_col.copy_from(&self.matrix_vector_prod(in_col));
86        }
87        res
88    }
89
90    /// Returns a copy of the diagonal elements as a `DVector`.
91    fn diagonal(&self) -> DVector<f64> {
92        DVector::from(self.diag.clone())
93    }
94
95    /// Sets the diagonal elements from a `DVector`.
96    fn set_diagonal(&mut self, diag: &DVector<f64>) {
97        for i in 0..self.n {
98            self.diag[i] = diag[i];
99        }
100    }
101
102    /// Returns the number of columns (matrix dimension).
103    fn ncols(&self) -> usize {
104        self.n
105    }
106
107    /// Returns the number of rows (matrix dimension).
108    fn nrows(&self) -> usize {
109        self.n
110    }
111}
