1//! # Variational HCI Module (`var`)
2//!
3//! This module implements the variational stage of the Heat-bath Configuration Interaction (HCI) algorithm.
4//! HCI is a selected CI method that iteratively builds a compact and important subset of the
5//! full CI space.
6//!
7//! ## Algorithm Overview:
8//! 1. **Initialization:** Start with a reference wavefunction (e.g., Hartree-Fock).
9//! 2. **Iteration:** Repeat until convergence:
10//!    a. **Search/Add:** Identify important determinants connected to the current variational
11//!       space but not yet included, based on a screening threshold (`eps_var`). Add these
12//!       new determinants to the variational space (`find_new_dets`).
13//!    b. **Diagonalize:** Construct the Hamiltonian matrix within the current (expanded)
14//!       variational space and solve the eigenvalue problem (using the Davidson algorithm
15//!       implemented in `davidson::sparse_optimize`) to obtain the updated variational
16//!       energy and wavefunction coefficients.
17//!    c. **Update Epsilon:** Decrease the screening threshold `eps_var` according to a schedule
18//!       (managed by `VarWf::eps_iter`).
19//!    d. **Check Convergence:** Stop if no new determinants are added at the target `eps_var`,
20//!       or if the energy converges, or if a maximum iteration count is reached.
21//!
22//! ## Submodules:
23//! *   `davidson`: Implementation of the Davidson diagonalization algorithm for sparse matrices.
24//! *   `eigenvalues`: (Potentially external or older) eigenvalue solver components.
25//! *   `ham_gen`: Functions for generating Hamiltonian matrix elements for the variational space.
26//! *   `off_diag`: Utilities for handling off-diagonal elements in the sparse matrix.
27//! *   `sparse`: Custom sparse matrix representations.
28//! *   `utils`: Helper functions specific to the variational stage.
29

49/// Performs the iterative variational Heat-bath Configuration Interaction (HCI) calculation.
50///
51/// Implements the main HCI loop: iteratively finding important connected determinants,
52/// adding them to the variational space (`var_wf`), and diagonalizing the Hamiltonian
53/// in the current space using the Davidson algorithm (`sparse_optimize`). The screening
54/// threshold `eps_var` is controlled by `var_wf.eps_iter`.
55///
56/// # Arguments
57/// * `global`: Global calculation parameters (e.g., `eps_var` target).
58/// * `ham`: The Hamiltonian operator.
59/// * `excite_gen`: Pre-computed excitation generator.
60/// * `var_wf`: Mutable reference to the variational wavefunction structure, which is
61///   updated in place throughout the iterations.
62pub fn variational(global: &Global, ham: &Ham, excite_gen: &ExciteGenerator, var_wf: &mut VarWf) {
63    let mut iter: i32 = 0;
64
65    println!(
66        "Start of variational stage: Wavefunction has {} det with energy {:.14}",
67        var_wf.wf.n, var_wf.wf.energy
68    );
69
70    // let eps_energy_converged: f64 = 2.5e-4;
71    let mut last_energy: Option<f64>;
72
73    while !var_wf.converged {
74        iter += 1;
75
76        if iter > 20 {
77            println!("Too many iterations! Stopping");
78            break;
79        }
80
81        let start_find_new_dets: Instant = Instant::now();
82        // --- Search/Add Step ---
83        // Find new determinants connected to the current space that are important
84        // according to the current screening threshold `var_wf.eps`.
85        let no_new_dets_added = var_wf.find_new_dets(global, ham, excite_gen); // Modifies var_wf.wf
86        println!("Time to find new dets: {:?}", start_find_new_dets.elapsed());
87
88        // Check for convergence: If we are at the target epsilon and no new determinants were added.
89        if var_wf.eps == global.eps_var && no_new_dets_added {
90            println!("No new dets added at target eps_var; variational stage converged.");
91            var_wf.converged = true;
92            break; // Exit the loop
93        }
94
95        last_energy = Some(var_wf.wf.energy);
96
97        let coeff_eps: f64 = 1e-4; // Davidson convergence epsilon for coefficients
98        let energy_eps: f64 = 1e-8; // Davidson convergence epsilon for energy
99
100        // --- Diagonalization Step ---
101        // Solve the eigenvalue problem Hc = Ec within the current variational space
102        // using the Davidson algorithm. Updates `var_wf.wf.energy` and `var_wf.wf.dets[*].coeff`.
103        println!("\nOptimizing coefficients of wf with {} dets", var_wf.wf.n);
104        let start_optimize_coeffs: Instant = Instant::now();
105        sparse_optimize( // Calls the Davidson solver
106            global,     // Pass global parameters (might be needed by ham_gen)
107            ham,
108            excite_gen,
109            var_wf,     // Passed mutably to update energy/coeffs and build sparse H
110            coeff_eps,  // Davidson convergence threshold for eigenvector change
111            energy_eps, // Davidson convergence threshold for eigenvalue change
112            iter > 1,   // Flag indicating whether to reuse previous guess vectors
113        );
114        println!(
115            "Time to optimize wf coefficients: {:?}",
116            start_optimize_coeffs.elapsed()
117        );
118        // Removed commented-out call to dense_optimize
119
120        println!("End of iteration {} (eps = {:.1e}): Wavefunction has {} determinants with energy {:.14}", iter, var_wf.eps, var_wf.wf.n, var_wf.wf.energy);
121        if var_wf.wf.n <= 10 {
122            var_wf.print();
123        } else {
124            var_wf.print_largest(10);
125        }
126
127        // if iter == 2 { panic!("Debug!") }
128
129        // --- Update Epsilon & Check Other Convergence ---
130        // Get the epsilon for the *next* iteration from the Eps iterator.
131        var_wf.eps = var_wf.eps_iter.next().unwrap_or(global.eps_var);
132
133        // Optional: Add energy convergence check here if desired
134        // if let Some(last_e) = last_energy {
135        //     if (var_wf.wf.energy - last_e).abs() < some_energy_threshold {
136        //         println!("Variational energy converged.");
137        //         var_wf.converged = true;
138        //         break;
139        //     }
140        // }
141    }
142}
