1//! # Fast Hamiltonian Construction (Milestone 1)
2//!
3//! This module implements the fast lookup-based Hamiltonian construction from Li et al.,
4//! J. Chem. Theory Comput. 2018, 14, 8, 4478–4489. It replaces the O(N²) determinant
5//! connection finding in `ham_gen.rs` with an O(N log N) approach using alpha string
6//! lookup tables.

16/// Type alias for alpha/beta strings (u128 bitstrings)
17pub type AlphaString = u128;
18pub type BetaString = u128;
19
20/// Manages the variational space for fast Hamiltonian construction.
21/// Implements the lookup-based algorithm from Li et al. 2018.
22pub struct VariationalSpace {
23    /// A vector containing all determinants (`D` in the paper).
24    pub determinants: Vec<Config>,
25
26    /// A vector of unique alpha strings (`α` in the paper).
27    pub alpha_strings: Vec<AlphaString>,
28
29    /// A hash map from an alpha string to its index in `alpha_strings`.
30    /// This is `iα(α)` from the paper.
31    pub alpha_to_index: HashMap<AlphaString, usize>,
32
33    /// A nested vector mapping an alpha string index to a list of
34    /// determinant indices that share that alpha string. This is `Dα(iα)`.
35    pub alpha_to_dets: Vec<Vec<usize>>,
36
37    /// Alpha string single excitation lookup table.
38    /// Maps alpha string to all alpha strings connected by single excitation.
39    pub alpha_singles: HashMap<AlphaString, Vec<AlphaString>>,
40
41    /// Beta string single excitation lookup table.
42    /// Maps beta string to all beta strings connected by single excitation.
43    pub beta_singles: HashMap<BetaString, Vec<BetaString>>,
44
45    /// Map from (alpha_string, beta_r1) to list of (det_index, beta_string)
46    /// Used for fast opposite-spin excitation finding.
47    alpha_beta_lookup: HashMap<(AlphaString, BetaString), Vec<(usize, BetaString)>>,
48}
49
50impl VariationalSpace {
51    /// Constructs a new `VariationalSpace` from a vector of determinants.
52    /// This function builds the fast lookup tables for O(N log N) connection finding.
53    ///
54    /// # Arguments
55    /// * `determinants` - The initial set of determinants in the variational space.
56    pub fn new(determinants: Vec<Config>) -> Self {
57        let mut alpha_strings = Vec::new();
58        let mut alpha_to_index = HashMap::new();
59        let mut alpha_to_dets_map: HashMap<AlphaString, Vec<usize>> = HashMap::new();
60
61        // Step 1: Group determinants by alpha string
62        for (det_idx, det) in determinants.iter().enumerate() {
63            let alpha = det.up;
64            
65            // Add determinant index to the alpha string's list
66            alpha_to_dets_map.entry(alpha).or_insert_with(Vec::new).push(det_idx);
67        }
68
69        // Step 2: Build alpha_strings and alpha_to_index from grouped data
70        for (alpha, det_indices) in alpha_to_dets_map.iter() {
71            let alpha_idx = alpha_strings.len();
72            alpha_strings.push(*alpha);
73            alpha_to_index.insert(*alpha, alpha_idx);
74        }
75
76        // Step 3: Build final alpha_to_dets using consistent indexing
77        let mut alpha_to_dets = vec![Vec::new(); alpha_strings.len()];
78        for (alpha, det_indices) in alpha_to_dets_map {
79            let alpha_idx = alpha_to_index[&alpha];
80            alpha_to_dets[alpha_idx] = det_indices;
81        }
82
83        // Step 4: Build single excitation lookup tables
84        let alpha_singles = Self::build_alpha_singles(&alpha_strings);
85        let beta_singles = Self::build_beta_singles(&determinants);
86        
87        // Step 5: Build alpha-beta lookup for fast opposite-spin excitation finding
88        let alpha_beta_lookup = Self::build_alpha_beta_lookup(&determinants);
89
90        VariationalSpace {
91            determinants,
92            alpha_strings,
93            alpha_to_index,
94            alpha_to_dets,
95            alpha_singles,
96            beta_singles,
97            alpha_beta_lookup,
98        }
99    }

101    /// Builds alpha string single excitation lookup table.
102    /// For each alpha string, finds all alpha strings connected by a single excitation.
103    fn build_alpha_singles(alpha_strings: &[AlphaString]) -> HashMap<AlphaString, Vec<AlphaString>> {
104        let mut alpha_singles = HashMap::new();
105        let mut alpha_r1_to_alphas: HashMap<AlphaString, Vec<AlphaString>> = HashMap::new();
106
107        // Build reverse lookup: (alpha - 1 electron) -> [alphas]
108        for &alpha in alpha_strings {
109            for occupied_orb in bits(alpha) {
110                let alpha_r1 = ibclr(alpha, occupied_orb);
111                alpha_r1_to_alphas.entry(alpha_r1).or_insert_with(Vec::new).push(alpha);
112            }
113        }
114
115        // Build forward lookup: alpha -> [connected alphas]
116        for &alpha in alpha_strings {
117            let mut connected_alphas = Vec::new();
118            for occupied_orb in bits(alpha) {
119                let alpha_r1 = ibclr(alpha, occupied_orb);
120                if let Some(connected_list) = alpha_r1_to_alphas.get(&alpha_r1) {
121                    for &connected_alpha in connected_list {
122                        if connected_alpha != alpha {
123                            connected_alphas.push(connected_alpha);
124                        }
125                    }
126                }
127            }
128            alpha_singles.insert(alpha, connected_alphas);
129        }
130
131        alpha_singles
132    }

134    /// Builds beta string single excitation lookup table.
135    /// For each beta string in the determinant set, finds all connected beta strings.
136    fn build_beta_singles(determinants: &[Config]) -> HashMap<BetaString, Vec<BetaString>> {
137        let mut beta_singles = HashMap::new();
138        let mut beta_r1_to_betas: HashMap<BetaString, Vec<BetaString>> = HashMap::new();
139        let mut unique_betas = std::collections::HashSet::new();
140
141        // Collect unique beta strings
142        for det in determinants {
143            unique_betas.insert(det.dn);
144        }
145
146        // Build reverse lookup: (beta - 1 electron) -> [betas]
147        for &beta in &unique_betas {
148            for occupied_orb in bits(beta) {
149                let beta_r1 = ibclr(beta, occupied_orb);
150                beta_r1_to_betas.entry(beta_r1).or_insert_with(Vec::new).push(beta);
151            }
152        }
153
154        // Build forward lookup: beta -> [connected betas]
155        for &beta in &unique_betas {
156            let mut connected_betas = Vec::new();
157            for occupied_orb in bits(beta) {
158                let beta_r1 = ibclr(beta, occupied_orb);
159                if let Some(connected_list) = beta_r1_to_betas.get(&beta_r1) {
160                    for &connected_beta in connected_list {
161                        if connected_beta != beta {
162                            connected_betas.push(connected_beta);
163                        }
164                    }
165                }
166            }
167            beta_singles.insert(beta, connected_betas);
168        }
169
170        beta_singles
171    }

173    /// Builds alpha-beta lookup table for fast opposite-spin excitation finding.
174    /// Maps (alpha_string, beta_r1) to list of (det_index, beta_string).
175    fn build_alpha_beta_lookup(determinants: &[Config]) -> HashMap<(AlphaString, BetaString), Vec<(usize, BetaString)>> {
176        let mut alpha_beta_lookup = HashMap::new();
177
178        for (det_idx, det) in determinants.iter().enumerate() {
179            let alpha = det.up;
180            let beta = det.dn;
181            
182            // For each way to remove one electron from beta string
183            for occupied_orb in bits(beta) {
184                let beta_r1 = ibclr(beta, occupied_orb);
185                let key = (alpha, beta_r1);
186                alpha_beta_lookup.entry(key).or_insert_with(Vec::new).push((det_idx, beta));
187            }
188        }
189
190        alpha_beta_lookup
191    }

193    /// Finds all determinants connected to a given source determinant by a
194    /// single or double excitation, using the fast lookup tables.
195    ///
196    /// # Arguments
197    /// * `source_det_idx` - The index of the source determinant in `self.determinants`.
198    /// * `wf` - The variational wavefunction (for index mapping).
199    /// * `ham` - The Hamiltonian operator.
200    /// * `n_stored_h` - Number of determinants for which H matrix elements are already stored.
201    ///
202    /// # Returns
203    /// A vector of tuples, where each tuple contains the index of a connected
204    /// determinant and the corresponding Hamiltonian matrix element.
205    pub fn find_connections_and_add_to_sparse_ham(
206        &self,
207        source_det_idx: usize,
208        wf: &mut VarWf,
209        ham: &Ham,
210        n_stored_h: usize,
211    ) {
212        let source_config = self.determinants[source_det_idx];
213        let source_alpha = source_config.up;
214        let source_beta = source_config.dn;
215
216        // Find opposite-spin double excitations
217        self.find_opposite_spin_connections(source_det_idx, source_alpha, source_beta, wf, ham, n_stored_h);
218
219        // Find same-spin double excitations
220        self.find_same_spin_connections(source_det_idx, source_alpha, source_beta, wf, ham, n_stored_h);
221    }

223    /// Finds opposite-spin double excitation connections using fast lookup tables.
224    /// Implements the algorithm from Li et al. 2018 for mixed excitations.
225    /// An opposite-spin double excitation requires alpha single + beta single = overall double.
226    fn find_opposite_spin_connections(
227        &self,
228        source_det_idx: usize,
229        source_alpha: AlphaString,
230        source_beta: BetaString,
231        wf: &mut VarWf,
232        ham: &Ham,
233        n_stored_h: usize,
234    ) {
235        // For opposite-spin double excitations: need alpha single-connected AND beta single-connected
236        if let Some(connected_alphas) = self.alpha_singles.get(&source_alpha) {
237            for &connected_alpha in connected_alphas {
238                if let Some(connected_betas) = self.beta_singles.get(&source_beta) {
239                    for &connected_beta in connected_betas {
240                        // Check if a determinant with (connected_alpha, connected_beta) exists
241                        let target_config = Config {
242                            up: connected_alpha,
243                            dn: connected_beta,
244                        };
245                        
246                        // Look up this configuration in the determinant list
247                        if let Some(&connected_det_idx) = wf.wf.inds.get(&target_config) {
248                            // Only add if at least one determinant index is new
249                            if source_det_idx >= n_stored_h || connected_det_idx >= n_stored_h {
250                                add_el(wf, ham, source_det_idx, connected_det_idx, None);
251                            }
252                        }
253                    }
254                }
255            }
256        }
257    }

259    /// Finds same-spin double excitation connections.
260    /// For same-spin doubles: either same alpha + beta double, or same beta + alpha double.
261    /// To avoid double counting, only check same alpha + beta double (the other direction 
262    /// will be found when processing the other determinant).
263    pub fn find_same_spin_connections(
264        &self,
265        source_det_idx: usize,
266        source_alpha: AlphaString,
267        source_beta: BetaString,
268        wf: &mut VarWf,
269        ham: &Ham,
270        n_stored_h: usize,
271    ) {
272        // Same alpha string: check for beta double excitations only
273        // (This avoids double-counting since we'll find the reverse when processing the other det)
274        if let Some(alpha_idx) = self.alpha_to_index.get(&source_alpha) {
275            for &connected_det_idx in &self.alpha_to_dets[*alpha_idx] {
276                if connected_det_idx != source_det_idx {
277                    let connected_config = self.determinants[connected_det_idx];
278                    let connected_beta = connected_config.dn;
279
280                    // Check if beta strings differ by double excitation
281                    if self.are_strings_double_connected(source_beta, connected_beta) {
282                        // Only add if at least one determinant index is new
283                        if source_det_idx >= n_stored_h || connected_det_idx >= n_stored_h {
284                            add_el(wf, ham, source_det_idx, connected_det_idx, None);
285                        }
286                    }
287                }
288            }
289        }
290
291        // For same beta + alpha double: only process if source_det_idx has a NEW determinant
292        // This ensures we don't double-count (since add_el handles i<j ordering automatically)
293        if source_det_idx >= n_stored_h {
294            for (other_alpha, other_alpha_idx) in &self.alpha_to_index {
295                if *other_alpha != source_alpha && self.are_strings_double_connected(source_alpha, *other_alpha) {
296                    for &connected_det_idx in &self.alpha_to_dets[*other_alpha_idx] {
297                        let connected_config = self.determinants[connected_det_idx];
298                        if connected_config.dn == source_beta {
299                            add_el(wf, ham, source_det_idx, connected_det_idx, None);
300                        }
301                    }
302                }
303            }
304        }
305    }

307    /// Checks if two beta strings are connected by a single excitation.
308    fn are_beta_strings_single_connected(&self, beta1: BetaString, beta2: BetaString) -> bool {
309        if let Some(connected_betas) = self.beta_singles.get(&beta1) {
310            connected_betas.contains(&beta2)
311        } else {
312            false
313        }
314    }

316    /// Checks if two strings (alpha or beta) are connected by a double excitation.
317    /// Two strings are double-connected if they differ by exactly 4 bits (2 removals + 2 additions).
318    /// This represents moving exactly 2 electrons to different orbitals.
319    pub fn are_strings_double_connected(&self, string1: u128, string2: u128) -> bool {
320        let diff = string1 ^ string2;
321        let n_diffs = diff.count_ones();
322        
323        // Must differ by exactly 4 bits (2 electrons removed, 2 electrons added)
324        if n_diffs != 4 {
325            return false;
326        }
327        
328        // Additional validation: ensure same number of electrons in both strings
329        if string1.count_ones() != string2.count_ones() {
330            return false;
331        }
332        
333        true
334    }

336    /// Follows the exact pattern of the original algorithm:
337    /// - Opposite-spin: process NEW alpha strings, find connections to ANY determinants
338    /// - Same-spin: process ALL alpha strings, find connections involving NEW determinants
339    pub fn find_all_connections_original_pattern(
340        &self,
341        wf: &mut VarWf,
342        ham: &Ham,
343        n_stored_h: usize,
344    ) {
345        // First find all NEW alpha strings (alpha strings that have at least one NEW determinant)
346        let mut new_alpha_strings = std::collections::HashSet::new();
347        for det_idx in n_stored_h..self.determinants.len() {
348            let alpha = self.determinants[det_idx].up;
349            new_alpha_strings.insert(alpha);
350        }
351
352        // SINGLE EXCITATIONS: Process NEW determinants for single excitations
353        self.find_single_excitations(wf, ham, n_stored_h);
354
355        // OPPOSITE-SPIN EXCITATIONS: Process NEW alpha strings only
356        for &new_alpha in &new_alpha_strings {
357            self.find_opposite_spin_for_alpha_string(new_alpha, wf, ham, n_stored_h);
358        }
359
360        // SAME-SPIN EXCITATIONS: Process ALL alpha strings, but only connections involving NEW dets
361        for &alpha in &self.alpha_strings {
362            self.find_same_spin_for_alpha_string(alpha, wf, ham, n_stored_h);
363        }
364    }

366    /// Find opposite-spin connections for a specific alpha string (following original pattern)
367    fn find_opposite_spin_for_alpha_string(
368        &self,
369        source_alpha: AlphaString,
370        wf: &mut VarWf,
371        ham: &Ham,
372        n_stored_h: usize,
373    ) {
374        // Find all alpha strings connected by single excitation
375        if let Some(connected_alphas) = self.alpha_singles.get(&source_alpha) {
376            // Get all determinants with the source alpha
377            if let Some(source_alpha_idx) = self.alpha_to_index.get(&source_alpha) {
378                let source_dets = &self.alpha_to_dets[*source_alpha_idx];
379                
380                for &connected_alpha in connected_alphas {
381                    // Get all determinants with the connected alpha
382                    if let Some(connected_alpha_idx) = self.alpha_to_index.get(&connected_alpha) {
383                        let connected_dets = &self.alpha_to_dets[*connected_alpha_idx];
384                        
385                        // Check all combinations for opposite-spin double excitations
386                        for &source_det_idx in source_dets {
387                            let source_beta = self.determinants[source_det_idx].dn;
388                            
389                            for &connected_det_idx in connected_dets {
390                                let connected_beta = self.determinants[connected_det_idx].dn;
391                                
392                                // Check if this is a valid opposite-spin double excitation
393                                if self.are_beta_strings_single_connected(source_beta, connected_beta) {
394                                    // Only add if at least one determinant is NEW
395                                    if source_det_idx >= n_stored_h || connected_det_idx >= n_stored_h {
396                                        add_el(wf, ham, source_det_idx, connected_det_idx, None);
397                                    }
398                                }
399                            }
400                        }
401                    }
402                }
403            }
404        }
405    }

407    /// Find same-spin connections for a specific alpha string (following original pattern)
408    fn find_same_spin_for_alpha_string(
409        &self,
410        alpha: AlphaString,
411        wf: &mut VarWf,
412        ham: &Ham,
413        n_stored_h: usize,
414    ) {
415        // Get all determinants with this alpha string
416        if let Some(alpha_idx) = self.alpha_to_index.get(&alpha) {
417            let alpha_dets = &self.alpha_to_dets[*alpha_idx];
418            
419            // Check if there are any NEW determinants with this alpha string
420            let has_new_dets = alpha_dets.iter().any(|&det_idx| det_idx >= n_stored_h);
421            
422            if has_new_dets {
423                // Find same-spin excitations within this alpha string (beta double excitations)
424                for (i, &det_idx1) in alpha_dets.iter().enumerate() {
425                    for &det_idx2 in &alpha_dets[i+1..] {
426                        // Ensure at least one is new (following original pattern)
427                        if det_idx1 >= n_stored_h || det_idx2 >= n_stored_h {
428                            let beta1 = self.determinants[det_idx1].dn;
429                            let beta2 = self.determinants[det_idx2].dn;
430                            
431                            // Check if this is a valid same-spin double excitation (beta)
432                            if self.are_strings_double_connected(beta1, beta2) {
433                                add_el(wf, ham, det_idx1, det_idx2, None);
434                            }
435                        }
436                    }
437                }
438                
439                // Also handle alpha double excitations: find connections to OTHER alpha strings
440                // that differ by exactly 2 electrons, and check if they have determinants 
441                // with the SAME beta strings as our NEW determinants
442                let new_det_indices: Vec<usize> = alpha_dets.iter()
443                    .filter(|&&det_idx| det_idx >= n_stored_h)
444                    .copied()
445                    .collect();
446                
447                for &other_alpha in &self.alpha_strings {
448                    if other_alpha != alpha && self.are_strings_double_connected(alpha, other_alpha) {
449                        if let Some(other_alpha_idx) = self.alpha_to_index.get(&other_alpha) {
450                            let other_alpha_dets = &self.alpha_to_dets[*other_alpha_idx];
451                            
452                            for &new_det_idx in &new_det_indices {
453                                let new_beta = self.determinants[new_det_idx].dn;
454                                
455                                // Find determinants with other_alpha and same beta
456                                for &other_det_idx in other_alpha_dets {
457                                    let other_beta = self.determinants[other_det_idx].dn;
458                                    if new_beta == other_beta {
459                                        add_el(wf, ham, new_det_idx, other_det_idx, None);
460                                    }
461                                }
462                            }
463                        }
464                    }
465                }
466            }
467        }
468    }

470    /// Find single excitations connecting NEW determinants to ALL determinants
471    fn find_single_excitations(
472        &self,
473        wf: &mut VarWf,
474        ham: &Ham,
475        n_stored_h: usize,
476    ) {
477        // For each NEW determinant, check ALL determinants for single excitations
478        for source_idx in n_stored_h..self.determinants.len() {
479            let source_config = self.determinants[source_idx];
480            
481            // Check all determinants (both old and new)
482            for target_idx in 0..self.determinants.len() {
483                if source_idx == target_idx {
484                    continue; // Skip self
485                }
486                
487                let target_config = self.determinants[target_idx];
488                
489                // Check if they differ by exactly one electron
490                let up_diff = (source_config.up ^ target_config.up).count_ones();
491                let dn_diff = (source_config.dn ^ target_config.dn).count_ones();
492                
493                let is_single_excitation = match (up_diff, dn_diff) {
494                    (2, 0) => true,  // Alpha single excitation
495                    (0, 2) => true,  // Beta single excitation
496                    _ => false,      // Not a single excitation
497                };
498                
499                if is_single_excitation {
500                    // Only add upper triangular elements
501                    if source_idx < target_idx {
502                        let elem = ham.ham_off_diag_no_excite(&source_config, &target_config);
503                        if elem.abs() > 1e-12 {
504                            wf.sparse_ham.off_diag[source_idx].push((target_idx, elem));
505                        }
506                    } else if target_idx < source_idx {
507                        let elem = ham.ham_off_diag_no_excite(&target_config, &source_config);
508                        if elem.abs() > 1e-12 {
509                            wf.sparse_ham.off_diag[target_idx].push((source_idx, elem));
510                        }
511                    }
512                }
513            }
514        }
515    }

518/// Fast Hamiltonian construction using the VariationalSpace lookup tables.
519/// This replaces the algorithm in `gen_sparse_ham_fast` with the O(N log N) approach.
520///
521/// # Arguments
522/// * `global` - Global calculation parameters.
523/// * `wf` - Mutable reference to the variational wavefunction.
524/// * `ham` - The Hamiltonian operator.
525pub fn gen_sparse_ham_fast_lookup(
526    global: &Global,
527    wf: &mut VarWf,
528    ham: &Ham,
529) {
530    println!("Starting fast Hamiltonian construction with lookup tables...");
531    let start_time = std::time::Instant::now();
532
533    // Extract determinant configurations for the VariationalSpace
534    let mut determinants = Vec::with_capacity(wf.wf.n);
535    for det in &wf.wf.dets {
536        determinants.push(det.config);
537    }
538
539    // Build the VariationalSpace with lookup tables
540    let setup_start = std::time::Instant::now();
541    let var_space = VariationalSpace::new(determinants);
542    println!("Lookup table setup time: {:?}", setup_start.elapsed());
543
544    // Initialize sparse Hamiltonian matrix if needed
545    if wf.n_stored_h() == 0 {
546        wf.new_sparse_ham();
547    } else {
548        wf.expand_sparse_ham_rows();
549    }
550
551    // Find connections using the same pattern as the original algorithm
552    let connection_start = std::time::Instant::now();
553    let n_stored_h = wf.n_stored_h();
554    
555    println!("Finding connections for {} new determinants (indices {} to {})", 
556             wf.wf.n - n_stored_h, n_stored_h, wf.wf.n - 1);
557    
558    // Follow the original algorithm pattern:
559    // 1. Opposite-spin: process NEW alpha strings to ANY determinants
560    // 2. Same-spin: process ALL alpha strings but only with NEW determinants involved
561    
562    var_space.find_all_connections_original_pattern(wf, ham, n_stored_h);
563    
564    println!("Connection finding time: {:?}", connection_start.elapsed());
565
566    // Sort and remove duplicates
567    wf.sparse_ham.sort_remove_duplicates();
568
569    // Update stored H count
570    wf.update_n_stored_h(wf.wf.n);
571
572    println!("Total fast Hamiltonian construction time: {:?}", start_time.elapsed());
573}

914            test_config(&[0, 2], &[0, 2]),  // |02⍺02β⟩ - opposite-spin double

919        let ham = mock_ham();

922        let mut total_connections = 0;
923        
924        // Count connections from each determinant
925        for i in 0..4 {
926            let initial_count = wf.sparse_ham.off_diag.iter().map(|v| v.len()).sum::<usize>();
927            var_space.find_connections_and_add_to_sparse_ham(i, &mut wf, &ham, n_stored_h);
928            let final_count = wf.sparse_ham.off_diag.iter().map(|v| v.len()).sum::<usize>();
929            total_connections += final_count - initial_count;
930        }
931        
932        // Should find connections between:
933        // 0-1 (alpha single), 0-2 (beta single), 0-3 (opposite-spin double)
934        // 1-2 (opposite-spin double), 1-3 (beta single), 2-3 (alpha single)
935        // Total: 6 unique connections, but each found twice = 12
936        // However, the sparse matrix only stores upper triangle, so expect 6
937        println!("Total connections found: {}", total_connections);
938        
939        // Remove duplicates by sorting and deduplicating the sparse matrix
940        wf.sparse_ham.sort_remove_duplicates();
941        let unique_connections = wf.sparse_ham.off_diag.iter().map(|v| v.len()).sum::<usize>();
