3# Davidson Diagonalization
4
5The Davidson method is suitable for diagonal-dominant symmetric matrices,
6that are quite common in certain scientific problems like [electronic
7structure](https://en.wikipedia.org/wiki/Electronic_structure). The Davidson
8method could be not practical for other kind of symmetric matrices.
9
10The current implementation uses a general davidson algorithm, meaning
11that it compute all the requested eigenvalues simultaneusly using a variable
12size block approach. The family of Davidson algorithm only differ in the way
13that the correction vector is computed.
14
15Available correction methods are:
16 * **DPR**: Diagonal-Preconditioned-Residue
17 * **GJD**: Generalized Jacobi Davidson
18
19*/
20

34/// Structure containing the initial configuration data
35struct Config {
36    method: DavidsonCorrection,
37    spectrum_target: SpectrumTarget,
38    tolerance: f64,
39    max_iters: usize,
40    max_search_space: usize,
41    init_dim: usize,   // Initial dimension of the subpace
42    update_dim: usize, // number of vector to add to the search space
43    reset_dim: usize,  // dimension to reset to once max_search_space is reached
44    energy_tolerance: f64,
45}
46impl Config {
47    /// Choose sensible default values for the davidson algorithm, where:
48    /// * `nvalues` - Number of eigenvalue/eigenvector pairs to compute
49    /// * `dim` - dimension of the matrix to diagonalize
50    /// * `method` - Either DPR or GJD
51    /// * `target` Lowest, highest or somewhere in the middle portion of the spectrum
52    /// * `tolerance` Numerical tolerance to reach convergence
53    fn new(
54        nvalues: usize,
55        dim: usize,
56        init_dim: usize,
57        method: DavidsonCorrection,
58        target: SpectrumTarget,
59        tolerance: f64,
60        energy_tolerance: f64,
61    ) -> Self {
62        let max_search_space = if nvalues + 10 < dim {
63            nvalues + 10
64        } else {
65            dim
66        };
67        let update_dim = nvalues;
68        Config {
69            method,
70            spectrum_target: target,
71            tolerance,
72            energy_tolerance,
73            max_iters: 50,
74            max_search_space,
75            init_dim,
76            update_dim,
77            reset_dim: nvalues,
78        }
79    }
80}

94    pub eigenvalues: DVector<f64>,
95    pub eigenvectors: DMatrix<f64>,

98impl Davidson {
99    /// The new static method takes the following arguments:
100    /// * `h` - A highly diagonal symmetric matrix
101    /// * `nvalues` - the number of eigenvalues/eigenvectors pair to compute
102    /// * `init` - the initial vector (optional; can be None)
103    /// * `method` Either DPR or GJD
104    /// * `spectrum_target` Lowest or Highest part of the spectrum
105    /// * `tolerance` numerical tolerance.
106    pub fn new<M: MatrixOperations>(
107        h: &M,
108        nvalues: usize,
109        init: Option<DMatrix<f64>>,
110        method: DavidsonCorrection,
111        spectrum_target: SpectrumTarget,
112        tolerance: f64,
113        energy_tolerance: f64,
114        ham: &Ham,
115        excite_gen: &ExciteGenerator,
116        wf: &Wf,
117    ) -> Result<Self, DavidsonError> {
118        // Initial configuration
119        let init_dim;
120        match init {
121            None => init_dim = 2,    // If no input wf, start from a subspace of size 2
122            Some(_) => init_dim = 1, // Else, just start from the input wf
123        }
124        let conf = Config::new(
125            nvalues,
126            h.nrows(),
127            init_dim,
128            method,
129            spectrum_target,
130            tolerance,
131            energy_tolerance,
132        );
133
134        // Initial subpace
135        let mut dim_sub = conf.init_dim;
136        // 1.1 Select the initial orthogonal subspace
137        // Uses diagonal to find the dim_sub basis states with the lowest diagonal elements
138        let mut basis = Self::generate_subspace(&h.diagonal(), &conf, init);
139
140        // 1.2 Select the correction to use
141        let corrector = CorrectionMethod::<M>::new(&h, conf.method);
142
143        // 2. Generate subpace matrix problem by projecting into the basis
144        let first_subspace = basis.columns(0, dim_sub);
145        let mut matrix_subspace = h.matrix_matrix_prod(first_subspace);
146        let mut matrix_proj = first_subspace.transpose() * &matrix_subspace;
147        let mut last_eig: Vec<f64> = vec![0.0f64; nvalues];
148
149        // Outer loop block Davidson schema
150        let mut result = Err(DavidsonError);
151        for i in 0..conf.max_iters {
152            let ord_sort = !matches!(conf.spectrum_target, SpectrumTarget::Highest);
153
154            let eig = sort_eigenpairs(SymmetricEigen::new(matrix_proj.clone()), ord_sort);
155
156            if i > 0 {
157                println!(
158                    "Davidson Iteration {}: energy = {:.6}",
159                    i, eig.eigenvalues[0]
160                );
161            }
162
163            // 4. Check for convergence
164            // 4.1 Compute the residues
165            let ritz_vectors = basis.columns(0, dim_sub) * eig.eigenvectors.columns(0, dim_sub);
166            let residues = Self::compute_residues(&ritz_vectors, &matrix_subspace, &eig);
167
168            // 4.2 Check Converge for each pair eigenvalue/eigenvector
169            let errors = DVector::<f64>::from_iterator(
170                nvalues,
171                residues
172                    .columns(0, nvalues)
173                    .column_iter()
174                    .map(|col| col.norm()),
175            );
176            // 4.3 Check if all eigenvalues/eigenvectors have converged
177            // println!("Errors: {}", errors);
178            // Check whether all coefficients and energies have converged:
179            if h.nrows() <= 2
180                || (errors.iter().all(|&x| x < conf.tolerance)
181                    && eig
182                        .eigenvalues
183                        .iter()
184                        .zip(last_eig.iter())
185                        .all(|(&x, &y)| (x - y).abs() < conf.energy_tolerance))
186            {
187                result = Ok(Self::create_results(
188                    &eig.eigenvalues,
189                    &ritz_vectors,
190                    nvalues,
191                ));
192                break;
193            }
194            for i in 0..nvalues {
195                last_eig[i] = eig.eigenvalues[i];
196            }
197            // 5. Update subspace basis set
198            // 5.1 Add the correction vectors to the current basis
199            if dim_sub + conf.update_dim <= conf.max_search_space {
200                let correction = corrector.compute_correction(
201                    &residues,
202                    &eig.eigenvalues,
203                    // &ritz_vectors,
204                    conf.update_dim,
205                    ham,
206                    excite_gen,
207                    wf,
208                );
209                // correction vector can be close to zero!
210                if correction.norm() < 1e-9 {
211                    println!("Davidson converged because correction is too small!");
212                    result = Ok(Self::create_results(
213                        &eig.eigenvalues,
214                        &ritz_vectors,
215                        nvalues,
216                    ));
217                    break;
218                }
219                update_subspace(&mut basis, correction, (dim_sub, dim_sub + conf.update_dim));
220
221                // 6. Orthogonalize the subspace
222                if !MGS::orthonormalize(&mut basis, dim_sub, dim_sub + conf.update_dim) {
223                    // New direction is not independent of previous ones
224                    println!("Davidson converged because new correction vector is redundant!");
225                    result = Ok(Self::create_results(
226                        &eig.eigenvalues,
227                        &ritz_vectors,
228                        nvalues,
229                    ));
230                    break;
231                }
232
233                // Update projected matrix
234                matrix_subspace = {
235                    let mut tmp = matrix_subspace.insert_columns(dim_sub, conf.update_dim, 0.0);
236                    let new_block = h.matrix_matrix_prod(basis.columns(dim_sub, conf.update_dim));
237                    let mut slice = tmp.columns_mut(dim_sub, conf.update_dim);
238                    slice.copy_from(&new_block);
239                    tmp
240                };
241
242                matrix_proj = {
243                    let new_dim = dim_sub + conf.update_dim;
244                    let new_subspace = basis.columns(0, new_dim);
245                    let mut tmp = DMatrix::<f64>::zeros(new_dim, new_dim);
246                    let mut slice = tmp.index_mut((..dim_sub, ..dim_sub));
247                    slice.copy_from(&matrix_proj);
248                    let new_block = new_subspace.transpose()
249                        * matrix_subspace.columns(dim_sub, conf.update_dim);
250                    let mut slice = tmp.index_mut((.., dim_sub..));
251                    slice.copy_from(&new_block);
252                    let mut slice = tmp.index_mut((dim_sub.., ..));
253                    slice.copy_from(&new_block.transpose());
254                    tmp
255                };
256                // update counter
257                dim_sub += conf.update_dim;
258
259            // 5.2 Otherwise reduce the basis of the subspace to the current
260            // correction
261            } else {
262                println!("Reducing the Krylov basis");
263                dim_sub = conf.reset_dim;
264                basis.fill(0.0);
265                update_subspace(&mut basis, ritz_vectors, (0, dim_sub));
266                // Update projected matrix
267                matrix_subspace = h.matrix_matrix_prod(basis.columns(0, dim_sub));
268                matrix_proj = basis.columns(0, dim_sub).transpose() * &matrix_subspace;
269            }
270            // Check number of iterations
271            if i > conf.max_iters {
272                break;
273            }
274        }
275
276        result
277    }

279    /// Extract the requested eigenvalues/eigenvectors pairs
280    fn create_results(
281        subspace_eigenvalues: &DVector<f64>,
282        ritz_vectors: &DMatrix<f64>,
283        nvalues: usize,
284    ) -> Davidson {
285        let eigenvectors = DMatrix::<f64>::from_iterator(
286            ritz_vectors.nrows(),
287            nvalues,
288            ritz_vectors.columns(0, nvalues).iter().cloned(),
289        );
290        let eigenvalues = DVector::<f64>::from_iterator(
291            nvalues,
292            subspace_eigenvalues.rows(0, nvalues).iter().cloned(),
293        );
294        Davidson {
295            eigenvalues,
296            eigenvectors,
297        }
298    }

300    /// Residue vectors
301    fn compute_residues(
302        ritz_vectors: &DMatrix<f64>,
303        matrix_subspace: &DMatrix<f64>,
304        eig: &SymmetricEigen<f64, Dynamic>,
305    ) -> DMatrix<f64> {
306        let dim_sub = eig.eigenvalues.nrows();
307        let lambda = {
308            let mut tmp = DMatrix::<f64>::zeros(dim_sub, dim_sub);
309            tmp.set_diagonal(&eig.eigenvalues);
310            tmp
311        };
312        let vs = matrix_subspace * &eig.eigenvectors;
313        let guess = ritz_vectors * lambda;
314        vs - guess
315    }

317    /// Generate initial orthonormal subspace
318    fn generate_subspace(
319        diag: &DVector<f64>,
320        conf: &Config,
321        init: Option<DMatrix<f64>>,
322    ) -> DMatrix<f64> {
323        match init {
324            None => {
325                // If no input vector, start with the two basis states with lowest diagonal element
326                DMatrix::<f64>::identity(diag.nrows(), conf.max_search_space)
327            }
328            Some(v) => {
329                // If input vector exists, start with it (and other low-energy basis states if needed)
330                let mut basis = DMatrix::<f64>::identity(diag.nrows(), conf.max_search_space);
331                update_subspace(&mut basis, v, (conf.init_dim - 1, conf.init_dim));
332                if conf.init_dim > 1 {
333                    MGS::orthonormalize(&mut basis, conf.init_dim - 1, conf.init_dim);
334                }
335                basis
336            }
337        }
338    }

341/// Structure containing the correction methods
342struct CorrectionMethod<'a, M>
343where
344    M: MatrixOperations,
345{
346    /// The initial target matrix
347    target: &'a M,
348    /// Method used to compute the correction
349    method: DavidsonCorrection,
350}
351
352impl<'a, M> CorrectionMethod<'a, M>
353where
354    M: MatrixOperations,
355{
356    fn new(target: &'a M, method: DavidsonCorrection) -> CorrectionMethod<'a, M> {
357        CorrectionMethod { target, method }
358    }

360    /// compute the correction vectors using either DPR or GJD
361    fn compute_correction(
362        &self,
363        residues: &DMatrix<f64>,
364        eigenvalues: &DVector<f64>,
365        // ritz_vectors: &DMatrix<f64>,
366        update_dim: usize,
367        ham: &Ham,
368        excite_gen: &ExciteGenerator,
369        wf: &Wf,
370    ) -> DMatrix<f64> {
371        match self.method {
372            DavidsonCorrection::DPR => {
373                self.compute_dpr_correction(residues, eigenvalues, update_dim)
374            }
375            DavidsonCorrection::GJD => {
376                self.compute_gjd_correction(/*residues, eigenvalues, ritz_vectors*/)
377            }
378            DavidsonCorrection::HPR(eps) => self.compute_hpr_correction(
379                residues,
380                eigenvalues,
381                update_dim,
382                ham,
383                excite_gen,
384                wf,
385                eps,
386            ),
387        }
388    }

390    /// Use the Diagonal-Preconditioned-Residue (DPR) method to compute the correction
391    fn compute_dpr_correction(
392        &self,
393        residues: &DMatrix<f64>,
394        eigenvalues: &DVector<f64>,
395        update_dim: usize,
396    ) -> DMatrix<f64> {
397        let d = self.target.diagonal();
398        let mut correction = DMatrix::<f64>::zeros(self.target.nrows(), residues.ncols());
399        for (k, lambda) in eigenvalues.iter().enumerate() {
400            if k == update_dim {
401                break;
402            }
403            let tmp = DVector::<f64>::repeat(self.target.nrows(), *lambda) - &d;
404            let rs = residues.column(k).component_div(&tmp);
405            correction.set_column(k, &rs);
406        }
407        correction
408    }
409
410    /// Use the Heatbath-Preconditioned-Residue (HPR) method to compute the correction:
411    /// Recall that the preconditioner (M - E) ^ {-1} multiplies the residual, and
412    /// Davidson converges faster if M is close to H. Of course, we can't have M = H because
413    /// inverting H is too expensive. Diagonal preconditioning chooses M to be the diagonal of H.
414    /// Here, we use Heat-bath to approximate (H - E) ^ {-1}:
415    /// Let H = D + O
416    /// (H - E) ^ {-1} = (D - E) ^ {-1} * (1 + O / (D - E)) ^ {-1}
417    /// ~ (D - E) ^ {-1} * (1 - O / (D - E)) + O(O^2)
418    /// = (D - E) ^ {-1} - (D - E) ^ {-1} * O * (D - E) ^ {-1}
419    /// = (1 - (D - E) ^ {-1} * O) * (D - E) ^ {-1}
420    /// So, we compute the usual Davidson preconditioner, then act on it with an approximation
421    /// to the second term in parentheses above
422    fn compute_hpr_correction(
423        &self,
424        residues: &DMatrix<f64>,
425        eigenvalues: &DVector<f64>,
426        update_dim: usize,
427        ham: &Ham,
428        excite_gen: &ExciteGenerator,
429        wf: &Wf,
430        eps: f64,
431    ) -> DMatrix<f64> {
432        let d = self.target.diagonal();
433        let mut correction = DMatrix::<f64>::zeros(self.target.nrows(), residues.ncols());
434        for (k, lambda) in eigenvalues.iter().enumerate() {
435            if k == update_dim {
436                break;
437            }
438            let tmp = DVector::<f64>::repeat(self.target.nrows(), *lambda) - &d;
439            let rs = residues.column(k).component_div(&tmp);
440            // rs is (D - E) ^ {-1} * residual
441            println!("DPR Correctioin: {:?}", rs);
442            println!("Performing HB...");
443            let scaled_eps = eps; // * (rs[0] / wf.dets[0].coeff).abs();
444            println!(
445                "Wf eps = {}, but using scaled eps = {} instead",
446                eps, scaled_eps
447            );
448            let mut hb_off_times_rs = DVector::from(
449                wf.approx_matmul_off_diag_variational_no_singles(&rs, ham, excite_gen, scaled_eps),
450            ); // Perform off-diagonal heat-bath here
451            println!("Done performing HB...");
452            println!("{:?}", hb_off_times_rs);
453            for i in 0..5 {
454                hb_off_times_rs[i] = 0.0;
455            }
456            let rs2 = hb_off_times_rs.component_div(&tmp);
457            println!("HPR Correction: {:?}", &rs - &rs2);
458            correction.set_column(k, &(rs - rs2)); // Add the DPR correction to the heat-bath off-diagonal correction to get the full correction
459        }
460        correction
461    }

495/// Update the subpace with new vectors
496fn update_subspace(basis: &mut DMatrix<f64>, vectors: DMatrix<f64>, range: (usize, usize)) {
497    let (start, end) = range;
498    let mut slice = basis.index_mut((.., start..end));
499    slice.copy_from(&vectors.columns(0, end - start));
500}
