16    /// Create a test determinant configuration
17    fn config(up_orbs: &[i32], dn_orbs: &[i32]) -> Config {
18        let mut up = 0u128;
19        let mut dn = 0u128;
20        
21        for &orb in up_orbs {
22            up = ibset(up, orb);
23        }
24        for &orb in dn_orbs {
25            dn = ibset(dn, orb);
26        }
27        
28        Config { up, dn }
29    }

60    /// Create a mock VarWf with given determinants
61    fn mock_var_wf(determinants: Vec<Config>) -> VarWf {
62        let n = determinants.len();
63        let mut dets = Vec::with_capacity(n);
64        let mut inds = HashMap::new();
65        
66        for (i, config) in determinants.iter().enumerate() {
67            dets.push(Det {
68                config: *config,
69                coeff: if i == 0 { 1.0 } else { 0.0 },
70                diag: Some(-1.0),
71            });
72            inds.insert(*config, i);
73        }
74        
75        let wf = Wf {
76            n,
77            dets,
78            inds,
79            energy: -1.0,
80        };
81        
82        let mut var_wf = VarWf::default();
83        var_wf.wf = wf;
84        var_wf.new_sparse_ham();
85        var_wf
86    }

112        let idx_12 = var_space.alpha_to_index[&alpha_12];
