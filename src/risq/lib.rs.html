1//! # Rust Implementation of Semistochastic Quantum chemistry (RISQ)
2//!
3//! A library for performing electronic structure calculations using advanced
4//! configuration interaction methods including Heat-bath CI (HCI) and
5//! Semistochastic HCI (SHCI).
6//!
7//! ## Features
8//!
9//! - Heat-bath Configuration Interaction (HCI)
10//! - Semistochastic Epstein-Nesbet Perturbation Theory  
11//! - Dynamic Semistochastic Full CI Quantum Monte Carlo (DS-FCIQMC)
12//! - Efficient determinant representation using bitstrings
13//! - Davidson eigenvalue solver with preconditioning
14//! - Alias sampling for importance sampling
15//!
16//! ## Usage
17//!
18//! ```rust,ignore
19//! use risq::context::RisqContext;
20//! use risq::wf::init_var_wf;
21//! use risq::var::variational;
22//! use risq::pt::perturbative;
23//!
24//! # fn main() -> Result<(), Box<dyn std::error::Error>> {
25//! // Initialize calculation context
26//! let mut context = RisqContext::from_files("in.json", "FCIDUMP")?;
27//!
28//! // Set up wavefunction
29//! let mut wavefunction = init_var_wf(&context)?;
30//!
31//! // Run variational calculation
32//! let var_energy = variational(&mut context, &mut wavefunction)?;
33//!
34//! // Run perturbative correction
35//! let pt2_energy = perturbative(&context, &wavefunction.wf)?;
36//!
37//! println!("Total energy: {:.10}", var_energy + pt2_energy);
38//! # Ok(())
39//! # }
40//! ```
