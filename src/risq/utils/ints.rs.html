1//! # Integral and Indexing Utilities (`utils::ints`)

3//! This module provides helper functions related to reading files (like FCIDUMP),

13/// Reads a file line by line and returns an iterator over the lines.
14///
15/// Opens the file specified by `filename`, wraps it in a `BufReader` for efficiency,
16/// and returns an iterator that yields each line as `io::Result<String>`.
17///
18/// # Arguments
19/// * `filename`: A type that can be converted into a `Path` (e.g., `&str`).
20///
21/// # Returns
22/// An `io::Result` containing an iterator over the lines of the file.
23pub fn read_lines<P>(filename: P) -> io::Result<io::Lines<io::BufReader<File>>>
24where
25    P: AsRef<Path>,
26{
27    let file = File::open(filename)?;
28    Ok(io::BufReader::new(file).lines())
29}

31/// Combines two orbital indices `p` and `q` into a single unique `usize` index.
32///
33/// Assumes `p` and `q` are 1-based indices (as in FCIDUMP).
34/// Uses triangular numbering: `index = max(i,j)*(max(i,j)-1)/2 + min(i,j)`, where `i=|p|, j=|q|`.
35/// This maps a pair `(p, q)` to a unique index suitable for accessing a flattened
36/// upper (or lower) triangular matrix storing symmetric two-index quantities (like h_pq).
37/// Ignores the sign of `p` and `q`.
38pub fn combine_2(p: i32, q: i32) -> usize {
39    let i = p.abs();
40    let j = q.abs();
41    if i > j {
42        ((i * (i - 1)) / 2 + j) as usize
43    } else {
44        ((j * (j - 1)) / 2 + i) as usize
45    }
46}
47
48/// Combines two `usize` indices `i` and `j` using triangular numbering.
49/// Internal helper function, assumes `i, j` are already 0-based or appropriately adjusted.
50fn combine_2_usize(i: usize, j: usize) -> usize {
51    if i > j {
52        (i * (i - 1)) / 2 + j
53    } else {
54        (j * (j - 1)) / 2 + i
55    }
56}

58/// Combines four orbital indices `p, q, r, s` into a single unique `usize` index.
59///
60/// Assumes `p, q, r, s` are 1-based indices.
61/// Uses nested triangular numbering: first combines `(p, q)` into `idx_pq` using `combine_2`,
62/// then combines `(r, s)` into `idx_rs` using `combine_2`, and finally combines
63/// `idx_pq` and `idx_rs` using `combine_2_usize`.
64/// This maps the four indices to a unique index suitable for accessing a flattened
65/// representation of the two-electron integrals (pq|rs), exploiting permutation symmetry.
66pub fn combine_4(p: i32, q: i32, r: i32, s: i32) -> usize {
67    combine_2_usize(combine_2(p, q), combine_2(r, s))
68}

70/// Calculates the permutation sign (+1 or -1) between two configurations (`det1`, `det2`)
71/// that differ by exactly one spin-orbital.
72///
73/// This sign is needed for the Slater-Condon rules. It's determined by the parity
74/// (even or odd number of electrons) between the differing orbital indices.
75/// Assumes `det1` and `det2` differ by a single bit flip.
76pub fn permute(det1: u128, det2: u128) -> i32 {
77    if det1 > det2 {
78        parity(det1 & (det1 - det2))
79    } else {
80        parity(det2 & (det2 - det1))
81    }
82}
83
84/// Calculates the permutation sign (+1 or -1) between two configurations (`det1`, `det2`)
85/// that differ by exactly two spin-orbitals.
86///
87/// `v` should contain the four orbital indices involved in the double excitation `(i, j -> k, l)`.
88/// The sign depends on the parity of electrons between `i` and `k` and between `j` and `l`.
89/// This implementation seems to calculate the parity based on the bits *between* the
90/// excited orbitals, which is a common way to determine the sign factor.
91/// Assumes `det1` and `det2` differ by exactly two bit flips corresponding to the indices in `v`.
92pub fn permute_2(det1: u128, det2: u128, v: [i32; 4]) -> i32 {
93    let diff: u128 = det1
94        & det2
95        & (((1 << v[0]) - 1) ^ ((1 << v[1]) - 1) ^ ((1 << v[2]) - 1) ^ ((1 << v[3]) - 1));
96    parity(diff)
97}
