5//! represented as `u128` bitstrings.

11/// Creates an iterator that yields the indices (0-based) of the set bits in `n`.
12///
13/// Example: `bits(6)` (binary `110`) would yield `1`, then `2`.
14pub fn bits(n: u128) -> impl Iterator<Item = i32> {
15    let mut bits_left = n;
16    std::iter::from_fn(move || {
17        if bits_left == 0 {
18            None
19        } else {
20            let res: i32 = bits_left.trailing_zeros() as i32;
21            bits_left &= !(1 << res);
22            Some(res)
23        }
24    })
25}
26
27/// Creates an iterator that yields unique pairs `(i, j)` where `i < j` and both
28/// bits `i` and `j` are set in `n`.
29///
30/// Example: `bits(7)` (binary `111`) would yield `(0, 1)`, `(0, 2)`, `(1, 2)`.
31pub fn bit_pairs(n: u128) -> impl Iterator<Item = (i32, i32)> {
32    let mut first_bit: i32 = n.trailing_zeros() as i32;
33    let mut first_bits_left: u128 = n & !(1 << first_bit);
34    let mut second_bits_left: u128 = first_bits_left;
35    std::iter::from_fn(move || {
36        if first_bits_left == 0 {
37            None
38        } else {
39            if second_bits_left == 0 {
40                first_bit = first_bits_left.trailing_zeros() as i32;
41                first_bits_left &= !(1 << first_bit);
42                if first_bits_left == 0 {
43                    return None;
44                };
45                second_bits_left = first_bits_left;
46            };
47            let second_bit: i32 = second_bits_left.trailing_zeros() as i32;
48            second_bits_left &= !(1 << second_bit);
49            Some((first_bit, second_bit))
50        }
51    })
52}

54/// Creates an iterator over all occupied spatial orbital indices in a determinant `det`,
55/// irrespective of spin. Combines the results of `bits(det.up)` and `bits(det.dn)`.
56pub fn det_bits(det: &Config) -> impl Iterator<Item = i32> {
57    bits(det.up).chain(bits(det.dn))
58}
59
60/// Creates an iterator over all pairs `(i, j)` where `i` is an occupied alpha orbital
61/// and `j` is an occupied beta orbital in the determinant `det`.
62/// Represents opposite-spin electron pairs.
63pub fn product_bits(det: &Config) -> impl Iterator<Item = (i32, i32)> {
64    let dn = det.dn;
65    bits(det.up).flat_map(move |i| bits(dn).map(move |j| (i, j)))
66}

68/// Creates an iterator yielding all single occupied valence orbitals and pairs of occupied
69/// valence orbitals within a determinant `det`.
70///
71/// Filters orbitals based on the `valence` mask in `excite_gen`.
72/// Yields tuples `(is_alpha, orbs)`, where `is_alpha` indicates the spin channel
73/// (Some(true) for alpha, Some(false) for beta, None for opposite-spin pairs) and
74/// `orbs` is the `Orbs::Single` or `Orbs::Double` representing the electron(s).
75/// This is useful for iterating through all possible starting points for excitations.
76pub fn valence_elecs_and_epairs(
77    det: &Config,
78    excite_gen: &ExciteGenerator,
79) -> impl Iterator<Item = (Option<bool>, Orbs)> {
80    let valence_det: Config = Config {
81        up: det.up & excite_gen.valence,
82        dn: det.dn & excite_gen.valence,
83    };
84    epairs(&valence_det).chain(elecs(&valence_det))
85}

87/// Creates an iterator yielding all single occupied orbitals in `det`.
88///
89/// Yields tuples `(Some(is_alpha), Orbs::Single(p))`, where `is_alpha` is true for alpha
90/// electrons and false for beta electrons.
91pub fn elecs(det: &Config) -> impl Iterator<Item = (Option<bool>, Orbs)> {
92    bits(det.up)
93        .map(|p| (Some(true), Orbs::Single(p)))
94        .chain(bits(det.dn).map(|p| (Some(false), Orbs::Single(p))))
95}

97/// Creates an iterator yielding single occupied *valence* orbitals in `det`.
98///
99/// Filters the output of `elecs` using the `valence` mask from `excite_gen`.
100/// Yields tuples `(Some(is_alpha), Orbs::Single(p))`.
101pub fn valence_elecs(
102    det: &Config,
103    excite_gen: &ExciteGenerator,
104) -> impl Iterator<Item = (Option<bool>, Orbs)> {
105    let valence_det: Config = Config {
106        up: det.up & excite_gen.valence,
107        dn: det.dn & excite_gen.valence,
108    };
109    elecs(&valence_det)
110}

112/// Creates an iterator yielding all pairs of occupied orbitals in `det`.
113///
114/// Includes both opposite-spin pairs (`(None, Orbs::Double(p_up, q_dn))`) generated
115/// via `product_bits`, and same-spin pairs (`(Some(spin), Orbs::Double(p, q))`)
116/// generated via `bit_pairs` for both alpha and beta electrons.
117pub fn epairs(det: &Config) -> impl Iterator<Item = (Option<bool>, Orbs)> {
118    product_bits(det).map(|pq| (None, Orbs::Double(pq))).chain(
119        bit_pairs(det.up)
120            .map(|pq| (Some(true), Orbs::Double(pq)))
121            .chain(bit_pairs(det.dn).map(|pq| (Some(false), Orbs::Double(pq)))),
122    )
123}

140/// Creates an iterator yielding all single occupied orbitals and all pairs of occupied orbitals.
141/// Combines the output of `epairs` and `elecs`.
142pub fn orbs(det: &Config) -> impl Iterator<Item = (Option<bool>, Orbs)> {
143    epairs(det).chain(
144        bits(det.up)
145            .map(|p| (Some(true), Orbs::Single(p)))
146            .chain(bits(det.dn).map(|p| (Some(false), Orbs::Single(p)))),
147    )
148}

152/// Sets the `b`-th bit (0-based) of `n` to 1. Equivalent to Fortran's `IBSET`.
153pub fn ibset(n: u128, b: i32) -> u128 {
154    n | (1 << b)
155}
156
157/// Clears the `b`-th bit (0-based) of `n` to 0. Equivalent to Fortran's `IBCLR`.
158pub fn ibclr(n: u128, b: i32) -> u128 {
159    n & !(1 << b)
160}
161
162/// Tests the `b`-th bit (0-based) of `n`. Returns `true` if the bit is 1, `false` otherwise.
163/// Equivalent to Fortran's `BTEST`.
164pub fn btest(n: u128, b: i32) -> bool {
165    !(n & (1 << b) == 0)
166}

168/// Calculates the parity of the number of set bits in `n` using a fast parallel algorithm.
169/// Returns `1` if the number of set bits is even, `-1` if it is odd.
170/// Useful for determining the sign factor when applying fermionic operators.
171pub fn parity(mut n: u128) -> i32 {
172    n ^= n >> 64;
173    n ^= n >> 32;
174    n ^= n >> 16;
175    n ^= n >> 8;
176    n ^= n >> 4;
177    n ^= n >> 2;
178    n ^= n >> 1;
179    1 - 2 * ((n & 1) as i32)
180}
