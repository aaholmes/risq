1//! # Input File Reading (`utils::read_input`)
2//!
3//! This module defines the `Global` struct which holds all calculation parameters
4//! read from the input file (typically `in.json`), and the `read_input` function
5//! which performs the file reading and deserialization using `serde_json`.
6
7extern crate serde;
8extern crate serde_json;
9
10use serde::Deserialize;
11use serde_json::{from_reader, Result};
12
13use std::fs::File;
14use std::io::BufReader;
15use std::path::Path;
16
17/// Holds global configuration parameters read from the input file (e.g., `in.json`).
18/// Uses `serde` to deserialize from JSON.
19#[derive(Deserialize, Debug)]
20pub struct Global {
21    /// Total number of spatial orbitals.
22    pub norb: i32,
23    /// Number of core orbitals to freeze (lowest energy based on h_ii).
24    pub norb_core: i32,
25    /// Number of alpha electrons.
26    pub nup: i32,
27    /// Number of beta electrons.
28    pub ndn: i32,
29    /// Target spin symmetry (+1 for singlet, -1 for triplet, etc. - affects initial epsilon).
30    pub z_sym: i32,
31    /// Number of electronic states to target
32    pub n_states: i32,
33    /// Variational screening threshold (epsilon_1) for HCI.
34    pub eps_var: f64,
35    /// Screening threshold dividing deterministic and stochastic parts in semistochastic PT.
36    pub eps_pt_dtm: f64,
37    /// Selector for opposite-spin algorithm variant (internal use).
38    pub opp_algo: i32,
39    /// Selector for same-spin algorithm variant (internal use).
40    pub same_algo: i32,
41    /// Target standard error for the stochastic PT energy component.
42    pub target_uncertainty: f64,
43    /// Number of stochastic samples to collect in each PT batch.
44    pub n_samples_per_batch: i32,
45    /// Maximum number of batches for stochastic PT (may terminate early if uncertainty target met).
46    pub n_batches: i32,
47    /// Number of samples for cross terms
48    pub n_cross_term_samples: i32,
49    /// Flag to select between different semistochastic PT implementations.
50    pub use_new_semistoch: bool,
51}
52
53/// Reads calculation parameters from a JSON file into a `Global` struct.
54///
55/// # Arguments
56/// * `path`: The path to the JSON input file (e.g., "in.json").
57///
58/// # Returns
59/// A `serde_json::Result<Global>` which is `Ok(Global)` on successful parsing,
60/// or `Err` if the file cannot be opened or the JSON is malformed/doesn't match
61/// the `Global` struct definition.
62///
63/// Also prints the parsed parameters to standard output.
64pub fn read_input<P: AsRef<Path>>(path: P) -> Result<Global> {
65    let file = File::open(path).unwrap();
66    let reader = BufReader::new(file);
67
68    let global: Global = from_reader(reader)?;
69
70    println!("\nInput:");
71    println!("  norb: {}", global.norb);
72    println!("  norb_core: {}", global.norb_core);
73    println!("  nup: {}", global.nup);
74    println!("  ndn: {}", global.ndn);
75    println!("  z_sym: {}", global.z_sym);
76    println!("  n_states: {}", global.n_states);
77    println!("  eps_var: {}", global.eps_var);
78    println!("  eps_pt_dtm: {}", global.eps_pt_dtm);
79    println!("  opp_algo: {}", global.opp_algo);
80    println!("  same_algo: {}", global.same_algo);
81    println!("  target_uncertainty: {}", global.target_uncertainty);
82    println!("  n_samples_per_batch: {}", global.n_samples_per_batch);
83    println!("  n_batches: {}", global.n_batches);
84    println!("  n_cross_term_samples: {}", global.n_cross_term_samples);
85    println!("  use_new_semistoch: {}", global.use_new_semistoch);
86    println!("\n");
87
88    // Removed commented-out validation checks. Consider re-adding if needed.
89
90    Ok(global)
91}
