12/// Holds the off-diagonal samples: (E_a,
13/// sum_i H_{ai} c_i w_i / p_i, sum_i (H_{ai} c_i w_i / p_i) ^ 2)
14/// for each PT det
15#[derive(Default)]
16struct OffDiagSamples {
17    pub n: i32,
18    pub pt_energies_and_sums: HashMap<Config, (f64, f64, f64)>,
19}

28    /// Add a (variational det, PT det) pair
29    /// For single excitations, first compares matrix element to eps (since
30    /// only an upper bound has been computed at this point)
31    fn add_var_and_pt(
32        &mut self,
33        var_det: &Det,
34        is_alpha: Option<bool>,
35        init_orbs: Orbs,
36        excite: &StoredExcite,
37        pt_config: Config,
38        w: i32,
39        prob: f64,
40        ham: &Ham,
41        eps: f64,
42    ) {
43        let h_ai: f64 = ham.ham_off_diag_no_excite(&var_det.config, &pt_config);
44
45        // If single excite, we only had max |h|, so may still need to reject here
46        match excite.target {
47            Orbs::Single(_) => {
48                if (h_ai * var_det.coeff).abs() < eps {
49                    return;
50                }
51            }
52            _ => {}
53        }
54        assert!(
55            (h_ai * var_det.coeff).abs() >= eps,
56            "Got an excitation smaller than eps!"
57        );
58
59        // x_{ai} = H_{ai} c_i w_i / p_i
60        let x_ai: f64 = h_ai * var_det.coeff * (w as f64) / prob;
61
62        match self.pt_energies_and_sums.get_mut(&pt_config) {
63            None => {
64                // compute diagonal element e_a
65                let e_a: f64 = var_det.new_diag_stored(ham, is_alpha, init_orbs, excite);
66                self.pt_energies_and_sums
67                    .insert(pt_config, (e_a, x_ai, x_ai * x_ai));
68            }
69            Some((_, s, s_sq)) => {
70                // update sum and sum_sq, but don't re-compute diagonal element
71                *s += x_ai;
72                *s_sq += x_ai * x_ai;
73            }
74        }
75    }
76
77    /// Add the PT samples corresponding to a new variational det (only for the off-diagonal term)
78    pub fn add_new_var_det(
79        &mut self,
80        wf: &Wf,
81        var_det: &Det,
82        w: i32,
83        prob: f64,
84        excite_gen: &ExciteGenerator,
85        ham: &Ham,
86        eps: Option<f64>,
87    ) {
88        self.n += w;
89        let mut eps_local: f64 = 1e-9; // This epsilon should be effectively zero, not the usual eps_var or eps_pt_dtm
90        if let Some(_e) = eps {
91            eps_local = eps.unwrap();
92        }
93        for (is_alpha, init_orbs, stored_excite) in excites(var_det, excite_gen, eps_local) {
94            let pt_config = var_det
95                .config
96                .apply_excite(is_alpha, &init_orbs, stored_excite);
97            if !wf.inds.contains_key(&pt_config) {
98                self.add_var_and_pt(
99                    var_det,
100                    is_alpha,
101                    init_orbs,
102                    stored_excite,
103                    pt_config,
104                    w,
105                    prob,
106                    ham,
107                    eps_local,
108                );
109            }
110        }
111    }
112
113    /// Compute the off-diagonal contribution to the PT energy estimate
114    /// using the stored samples
115    pub fn off_diag_pt_energy(&self, e0: f64) -> f64 {
116        let mut e_pt: f64 = 0.0;
117        for (e_a, s, s_sq) in self.pt_energies_and_sums.values() {
118            e_pt += (*s * *s - *s_sq) / (e0 - e_a);
119        }
120        e_pt / (self.n as f64) / ((self.n - 1) as f64)
121    }
122}

124/// Sample the 'off-diagonal' contribution to the PT energy (i.e., the term coming from the off-diagonal of
125/// the matrix V \[1 / (E_0 - E_a)\] V), and update the Welford statistics for the energy
126/// If eps is not None, then evaluate at eps=0 and subtract eps=eps.unwrap()
127pub fn sample_off_diag_update_welford(
128    wf: &Wf,
129    excite_gen: &ExciteGenerator,
130    ham: &Ham,
131    det_sampler: &Alias,
132    n_samples_per_batch: i32,
133    rand: &mut Rand,
134    enpt2_off_diag: &mut Stats<f64>,
135    eps: Option<f64>,
136) {
137    // Sample i with probability proportional to |c_i| max_a |H_{ai}|
138    // Note that this is the max over all PT dets, not just those left over after deterministic component!
139
140    let mut counts_and_probs: HashMap<usize, (i32, f64)> = HashMap::default();
141    for _i_sample in 0..n_samples_per_batch {
142        // Sample a variational det i, and add it to the samples data structure
143        let (i, p_i) = det_sampler.sample_with_prob(rand);
144        match counts_and_probs.get_mut(&i) {
145            None => {
146                counts_and_probs.insert(i, (1, p_i));
147            }
148            Some((w, _)) => {
149                *w += 1;
150            }
151        }
152    }
153
154    // Obtain var_dets and their counts {w}
155    let mut off_diag: OffDiagSamples = OffDiagSamples::default();
156    let mut off_diag_screened: OffDiagSamples = OffDiagSamples::default();
157    for (i, (w_i, p_i)) in counts_and_probs {
158        off_diag.add_new_var_det(wf, &wf.dets[i], w_i, p_i, excite_gen, ham, None);
159        if let Some(_e) = eps {
160            off_diag_screened.add_new_var_det(wf, &wf.dets[i], w_i, p_i, excite_gen, ham, eps);
161        }
162    }
163    assert_eq!(n_samples_per_batch, off_diag.n);
164
165    let mut off_diag_estimate: f64 = off_diag.off_diag_pt_energy(wf.energy);
166    if let Some(_e) = eps {
167        off_diag_estimate -= off_diag_screened.off_diag_pt_energy(wf.energy);
168    }
169    println!(
170        "Sampled off-diagonal energy this batch: {:.4}",
171        off_diag_estimate
172    );
173    enpt2_off_diag.update(off_diag_estimate);
174}
