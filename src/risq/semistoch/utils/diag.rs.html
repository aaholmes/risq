7/// Sample the 'diagonal' contribution to the PT energy (i.e., the term coming from the diagonal of
8/// the matrix V \[1 / (E_0 - E_a)\] V), and update the Welford statistics for the energy
9pub fn sample_diag_update_welford(
10    screened_sampler: &ScreenedSampler,
11    excite_gen: &ExciteGenerator,
12    ham: &Ham,
13    rand: &mut Rand,
14    e0: f64,
15    enpt2_diag: &mut Stats<f64>,
16) {
17    // Sample with probability proportional to (Hc)^2
18    let (sampled_det_info, sampled_prob) = matmul_sample_remaining(
19        screened_sampler,
20        ImpSampleDist::HcSquared,
21        excite_gen,
22        ham,
23        rand,
24    );
25
26    match sampled_det_info {
27        None => {
28            // Sampled excitation not valid
29        }
30        Some((exciting_det, excite, _target_det)) => {
31            // println!(
32            //     "Sampled excitation: Sampled det = {}, Sample prob = {}, (H_ai c_i)^2 / p = {}",
33            //     target_det,
34            //     sampled_prob,
35            //     excite.abs_h * excite.abs_h * exciting_det.coeff * exciting_det.coeff
36            //         / sampled_prob
37            // );
38            let e_a: f64 = exciting_det.new_diag(ham, &excite);
39            let energy: f64 = excite.abs_h * excite.abs_h * exciting_det.coeff * exciting_det.coeff
40                / (e0 - e_a)
41                / sampled_prob;
42            enpt2_diag.update(energy);
43        }
44    }
45}
