1//! # Alias Method Sampling (`stoch::alias`)
2//!
3//! Implements Vose's Alias Method for efficient O(1) sampling from a discrete
4//! probability distribution.
5//!
6//! This method pre-computes two tables (`alias` and `alias_prob`) based on the input
7//! probability distribution. Sampling then involves:
8//! 1. Rolling a fair die (selecting a random index `i` uniformly).
9//! 2. Flipping a biased coin (comparing a uniform random number `r` in [0,1) to `alias_prob[i]`).
10//! 3. If `r < alias_prob[i]`, return `i`. Otherwise, return `alias[i]`.
11//!
12//! This implementation is based on the description at [https://www.keithschwarz.com/darts-dice-coins/](https://www.keithschwarz.com/darts-dice-coins/)
13//! and inspired by the `vose-alias` crate, but uses vectors directly.
14
15use rand::Rng;
16use std::fmt::Debug;
17use crate::rng::Rand;
18use rand::distributions::Uniform;
19
20/// Data structure for Vose's Alias Method.
21///
22/// Contains the pre-computed tables needed for O(1) sampling.
23#[derive(Debug, Clone)]
24pub struct Alias {
25    /// The original normalized probability of sampling each element `i`. Stored for reference.
26    pub sample_prob: Vec<f64>,
27    /// The alias table. `alias[i]` stores the index of the alternative element
28    /// that might be chosen when column `i` is selected.
29    alias: Vec<usize>,
30    /// The probability table. `alias_prob[i]` stores the probability of choosing
31    /// index `i` (rather than `alias[i]`) when column `i` is selected.
32    alias_prob: Vec<f64>,
33    /// A distribution for sampling column indices uniformly from `0..size`.
34    uniform: Uniform<usize>,
35}
36
37impl Alias {
38    /// Creates a new `Alias` instance from a vector of relative probabilities.
39    ///
40    /// # Arguments
41    /// * `rel_probs`: A vector where `rel_probs[i]` is proportional to the desired
42    ///   probability of sampling index `i`. These probabilities do not need to be normalized.
43    ///
44    /// # Returns
45    /// An `Alias` struct initialized with the pre-computed tables.
46    ///
47    /// # Algorithm
48    /// Implements Vose's algorithm:
49    /// 1. Normalizes the input probabilities (`rel_probs`).
50    /// 2. Scales probabilities by `size` (number of elements).
51    /// 3. Initializes `small` and `large` worklists containing indices with scaled probabilities
52    ///    less than 1.0 and greater than or equal to 1.0, respectively.
53    /// 4. Iteratively processes pairs from `small` and `large`:
54    ///    - Sets `alias_prob[l]` for the small index `l`.
55    ///    - Sets `alias[l]` to the large index `g`.
56    ///    - Updates the remaining probability "mass" for the large index `g`.
57    ///    - Re-categorizes `g` into `small` or `large` based on its updated probability.
58    /// 5. Handles any remaining elements in `large` or `small` (due to numerical precision)
59    ///    by setting their `alias_prob` to 1.0.
60    pub fn new(rel_probs: Vec<f64>) -> Alias {
61        let size = rel_probs.len();
62        println!("New Alias with size = {}", size);
63
64        // Normalize input probabilities
65        let mut sum = 0.0;
66        for p in &rel_probs {
67            sum = sum + p;
68        }
69        let mut norm_prob: Vec<f64> = Vec::with_capacity(size);
70        for p in &rel_probs {
71            norm_prob.push(p / sum);
72        }
73
74        // starting the actual init
75        let mut small: Vec<usize> = Vec::with_capacity(size);
76        let mut large: Vec<usize> = Vec::with_capacity(size);
77        let mut scaled_probability_vector: Vec<f64> = Vec::with_capacity(size);
78
79        let mut alias: Vec<usize> = vec![0; size];
80        let mut prob: Vec<f64> = vec![0.0; size];
81
82        // multiply each prob by size
83        for i in 0..size {
84            let p: f64 = norm_prob[i];
85            let scaled_proba = p * (size as f64);
86            scaled_probability_vector.push(scaled_proba);
87
88            if scaled_proba < 1.0 {
89                small.push(i);
90            } else {
91                large.push(i);
92            }
93        }
94
95        // emptying one column first
96        while !(small.is_empty() || large.is_empty()) {
97            // removing the element from small and large
98            if let (Some(l), Some(g)) = (small.pop(), large.pop()) {
99                // put g in the alias vector
100                alias[l] = g;
101                // getting the probability of the small element
102                let p_l = scaled_probability_vector[l];
103                // put it in the prob vector
104                prob[l] = p_l;
105
106                // update the probability for g
107                let p_g = scaled_probability_vector[g];
108                // P(l->g) = 1 - p_l, so subtract this from p_g to get remaining prob to still be accounted for
109                let new_p_g = (p_g + p_l) - 1.0;
110                // update scaled_probability_vector
111                scaled_probability_vector[g] = new_p_g;
112                if new_p_g < 1.0 {
113                    small.push(g);
114                } else {
115                    large.push(g);
116                }
117            }
118        }
119
120        // finishing the init
121        while !large.is_empty() {
122            if let Some(g) = large.pop() {
123                prob[g] = 1.0;
124            }
125        }
126
127        while !small.is_empty() {
128            if let Some(l) = small.pop() {
129                prob[l] = 1.0;
130            }
131        }
132
133        Alias {
134            sample_prob: norm_prob,
135            alias,
136            alias_prob: prob,
137            uniform: Uniform::from(0..size),
138        }
139    }
140
141    /// Samples an index from the distribution in O(1) time.
142    ///
143    /// # Arguments
144    /// * `rand`: A mutable reference to the `Rand` struct containing the RNG state.
145    ///
146    /// # Returns
147    /// The index (`usize`) of the sampled element.
148    pub fn sample(&self, rand: &mut Rand) -> usize { // Made immutable self
149        let (i, r) = self.roll_die_and_flip_coin(rand);
150        return self.select_element(i, r);
151    }
152
153    /// Samples an index and returns it along with its true probability from the original distribution.
154    /// O(1) time complexity.
155    ///
156    /// # Arguments
157    /// * `rand`: A mutable reference to the `Rand` struct containing the RNG state.
158    ///
159    /// # Returns
160    /// A tuple `(index, probability)`, where `index` is the sampled index and `probability`
161    /// is its corresponding value from the original normalized distribution (`self.sample_prob`).
162    pub fn sample_with_prob(&self, rand: &mut Rand) -> (usize, f64) {
163        let (i, r) = self.roll_die_and_flip_coin(rand);
164        return self.select_element_and_prob(i, r);
165    }
166
167    /// Performs the two random steps of the Alias method: selecting a uniform column index
168    /// and generating a uniform random float for comparison.
169    /// Internal helper function.
170    fn roll_die_and_flip_coin(&self, rand: &mut Rand) -> (usize, f64) {
171        let i: usize = rand.rng.sample(self.uniform);
172        let r: f64 = rand.rng.gen();
173        return (i, r);
174    }
175
176    /// Selects the final element based on the die roll and coin flip.
177    /// Internal helper function.
178    fn select_element(&self, die: usize, coin: f64) -> usize {
179        // choose randomly an element from the element vector
180        return if coin < self.alias_prob[die] {
181            die
182        } else {
183            self.alias[die]
184        };
185    }
186
187    /// Selects the final element and retrieves its original probability.
188    /// Internal helper function.
189    fn select_element_and_prob(&self, die: usize, coin: f64) -> (usize, f64) {
190        // choose randomly an element from the element vector
191        return if coin < self.alias_prob[die] {
192            (die, self.sample_prob[die])
193        } else {
194            let alias = self.alias[die];
195            (alias, self.sample_prob[alias])
196        };
197    }
198
199    /// Tests the sampler by drawing many samples and comparing frequencies to target probabilities.
200    /// Prints the comparison to standard output.
201    pub fn test(&self, n_samples: i32, rand: &mut Rand) { // Made immutable self
202        // Take n_samples samples, show frequency of samples vs probability for full distribution
203        let n = self.alias.len();
204        let mut freq: Vec<i32> = vec![0; n];
205        for _i_sample in 0..n_samples {
206            let sample = self.sample(rand);
207            freq[sample] += 1;
208        }
209        println!("Target prob, sampled prob, element");
210        for i in 0..n {
211            println!(
212                "{:.6},   {:.6},   {}",
213                self.sample_prob[i],
214                (freq[i] as f64) / (n_samples as f64),
215                i
216            );
217        }
218    }
