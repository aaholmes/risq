1//! # Stochastic Sampling Utilities (`stoch`)
2//!
3//! This module provides general utilities and data structures for performing
4//! stochastic sampling, particularly importance sampling, as required by
5//! semistochastic methods.
6//!
7//! ## Key Components:
8//! *   `alias`: Submodule implementing Alias sampling for efficient O(1) sampling from
9//!     discrete distributions.
10//! *   `utils`: Submodule containing other sampling helpers like CDF sampling.
11//! *   `ImpSampleDist`: Enum to specify the desired importance sampling distribution
12//!     (e.g., proportional to |Hc| or (Hc)^2).
13//! *   `DetOrbSample`: Represents a potential starting point for sampling an excitation
14//!     (a specific determinant and initial orbital(s)).
15//! *   `ScreenedSampler`: A structure holding pre-computed Alias tables for efficiently
16//!     sampling `DetOrbSample` instances according to different distributions.
17//! *   `matmul_sample_remaining`: Function to perform a full importance sampling step,
18//!     first sampling a `DetOrbSample` and then sampling a target excitation.
19

32/// Specifies the desired probability distribution for importance sampling.
33#[derive(Clone, Copy, Debug)] // Added Debug
34pub enum ImpSampleDist {
35    /// Sample with probability proportional to `|H_ai * c_i|`.
36    AbsHc,
37    /// Sample with probability proportional to `(H_ai * c_i)^2`.
38    HcSquared,
39}

41/// For importance sampling the component of the matmul that is screened out by the eps threshold
42
43/// Matmul_sample_remaining performs the whole excitation sampling (exciting pair and target), but
44/// this contains only data structures sampling the exciting det and its exciting electron pair;
45/// sampling the target electron pair uses CDF searching which only requires ExciteGenerator
46
47/// Lifetime 'a must last as long as the vector being semistochastically multiplied, since this
48/// struct has pointers to its components
49
50/// For doubles:
51/// Contains two samplers, for sampling with p ~ |Hc| and with p ~ (Hc)^2
52
53/// For singles:
54/// If uniform_singles is true, enables uniform sampling of the singles from small-magnitude
55/// determinants that were not already treated deterministically
56/// Else, singles are sampled alongside doubles using max |H| instead of |H| as their
57/// relative prob
58
59/// Also, contains elements, the list of elements being sampled (e.g. det/orb pairs)
60/// Facilitates efficient importance sampling of determinant/initial-orbital pairs.
61///
62/// This structure is generated from a list of `DetOrbSample` instances, which represent
63/// all possible starting points (determinant `i` and initial orbitals `init`) for excitations
64/// that fall below the deterministic screening threshold (`eps_pt_dtm`).
65///
66/// It pre-computes Alias tables (`det_orb_sampler_*`) allowing for O(1) sampling of a
67/// `DetOrbSample` according to either the `|H_ai * c_i|` or `(H_ai * c_i)^2` distribution
68/// (summed over potential targets `a`).
69#[derive(Debug)]
70pub struct ScreenedSampler<'a> {
71    /// Vector storing the actual `DetOrbSample` elements that can be sampled.
72    /// The index corresponds to the sample returned by the Alias samplers.
73    pub elements: Vec<DetOrbSample<'a>>,
74    /// Alias sampler for drawing samples with probability proportional to `sum_a |H_ai * c_i|`.
75    pub det_orb_sampler_abs_hc: Alias,
76    /// Alias sampler for drawing samples with probability proportional to `sum_a (H_ai * c_i)^2`.
77    pub det_orb_sampler_hc_squared: Alias,
78    /// The global sum of `sum_abs_hc` over all elements, used for normalization or total estimates.
79    pub sum_abs_hc_all_dets_orbs: f64,
80    /// The global sum of `sum_hc_squared` over all elements, used for normalization or total estimates.
81    pub sum_hc_squared_all_dets_orbs: f64,
82}
83
84impl ScreenedSampler<'_> {
85    /// Returns `true` if there are no determinant/orbital pairs to sample from.
86    ///
87    /// This happens when every screened contribution was handled deterministically
88    /// (all `sum_abs_h == 0`), in which case the stochastic correction is exactly zero.
89    pub fn is_empty(&self) -> bool {
90        self.elements.is_empty()
91    }
92}

94/// Represents a potential starting point for sampling an excitation.
95///
96/// Contains a reference to a source determinant (`det`) and the specific initial
97/// orbital(s) (`init`) and spin (`is_alpha`) from which an excitation originates.
98/// It also stores pre-calculated sums of Hamiltonian estimates (`sum_*_h`) and
99/// wavefunction-weighted estimates (`sum_*_hc`) over all possible *target* excitations
100/// originating from this specific `det`/`init`/`is_alpha` combination that fall *below*
101/// the deterministic threshold (`eps_pt_dtm`). These sums are used to build the
102/// `ScreenedSampler`'s Alias tables.
103#[derive(Clone, Copy, Debug)]
104pub struct DetOrbSample<'a> {
105    /// Reference to the source determinant (`i`). Lifetime `'a` ensures it lives as long as the source `Wf`.
106    pub det: &'a Det,
107    /// The initial orbital(s) the excitation originates from.
108    pub init: Orbs,
109    /// The spin channel of the excitation (None for opposite-spin doubles).
110    pub is_alpha: Option<bool>,
111    /// Sum of `|H_ai|` over all target excitations `a` originating from this `det`/`init`/`is_alpha`
112    /// that were screened out (i.e., below `eps_pt_dtm`).
113    pub sum_abs_h: f64,
114    /// Sum of `H_ai^2` over all target excitations `a` originating from this `det`/`init`/`is_alpha`
115    /// that were screened out.
116    pub sum_h_squared: f64,
117    /// Sum of `|H_ai * c_i|` over all target excitations `a` originating from this `det`/`init`/`is_alpha`
118    /// that were screened out. Used for `AbsHc` importance sampling distribution.
119    pub sum_abs_hc: f64,
120    /// Sum of `(H_ai * c_i)^2` over all target excitations `a` originating from this `det`/`init`/`is_alpha`
121    /// that were screened out. Used for `HcSquared` importance sampling distribution.
122    pub sum_hc_squared: f64,
123}
124
125impl PartialEq for DetOrbSample<'_> {
126    /// Equality based on determinant configuration, initial orbitals, and spin channel.
127    fn eq(&self, other: &Self) -> bool {
128        self.det.config == other.det.config // Use Config's PartialEq
129            && self.init == other.init
130            && self.is_alpha == other.is_alpha
131    }
132}
133impl Eq for DetOrbSample<'_> {}
134
135impl Hash for DetOrbSample<'_> {
136    /// Hashes based on determinant configuration, initial orbitals, and spin channel. Consistent with `PartialEq`.
137    fn hash<H: Hasher>(&self, state: &mut H) {
138        self.det.config.hash(state);
139        self.init.hash(state);
140        self.is_alpha.hash(state);
141    }
142}

144/// Creates a `ScreenedSampler` from a vector of `DetOrbSample` instances.
145///
146/// Calculates the total sums (`sum_*_all_dets_orbs`) and constructs the Alias tables
147/// (`det_orb_sampler_*`) based on the normalized `sum_abs_hc` and `sum_hc_squared` values
148/// from the input `det_orbs`. Filters out any `DetOrbSample` with zero probability (`sum_abs_h == 0.0`).
149///
150/// # Arguments
151/// * `det_orbs`: A vector containing all `DetOrbSample` instances representing the space to be sampled.
152///
153/// # Returns
154/// An initialized `ScreenedSampler` ready for O(1) sampling.
155pub fn generate_screened_sampler(det_orbs: Vec<DetOrbSample>) -> ScreenedSampler {
156    // pub fn generate_screened_sampler<'a>(eps: f64, det_orbs: &'a Vec<DetOrbSample>) -> ScreenedSampler<'a> {
157
158    println!("Generating screened sampler of size {}", det_orbs.len());
159
160    // Normalize probs (both |Hc| and (Hc)^2)
161    let mut det_orbs_nonzero: Vec<DetOrbSample> = Vec::with_capacity(det_orbs.len());
162    let mut probs_abs_hc: Vec<f64> = Vec::with_capacity(det_orbs.len());
163    let mut probs_hc_squared: Vec<f64> = Vec::with_capacity(det_orbs.len());
164
165    let sum_abs_hc_all_dets_orbs: f64 = det_orbs
166        .iter()
167        .fold(0f64, |sum, &val| sum + val.sum_abs_hc as f64);
168    let sum_hc_squared_all_dets_orbs: f64 = det_orbs
169        .iter()
170        .fold(0f64, |sum, &val| sum + val.sum_hc_squared as f64);
171
172    for val in det_orbs.iter() {
173        if val.sum_abs_h != 0.0 {
174            det_orbs_nonzero.push(*val);
175            probs_abs_hc.push(val.sum_abs_hc / sum_abs_hc_all_dets_orbs);
176            probs_hc_squared.push(val.sum_hc_squared / sum_hc_squared_all_dets_orbs);
177        }
178    }
179
180    println!("Done generating screened sampler of size {}", det_orbs.len());
181
182    ScreenedSampler {
183        elements: det_orbs_nonzero,
184        det_orb_sampler_abs_hc: Alias::new(probs_abs_hc),
185        det_orb_sampler_hc_squared: Alias::new(probs_hc_squared),
186        sum_abs_hc_all_dets_orbs,
187        sum_hc_squared_all_dets_orbs,
188    }
189}

191/// Performs one full importance sampling step for the screened (stochastic) part of H*psi.
192///
193/// This function combines two sampling steps:
194/// 1. **Sample Source:** Samples a `DetOrbSample` (representing source determinant `i` and
195///    initial orbitals `init`) from the `screened_sampler` using Alias sampling according
196///    to the specified `imp_sample_dist` (`|Hc|` or `(Hc)^2`). This gives `det_orb_sample`
197///    and its sampling probability `det_orb_prob`.
198/// 2. **Sample Target:** Samples a specific target excitation (`StoredExcite`) from the list
199///    associated with the chosen `det_orb_sample` (using `excite_gen`). This step uses
200///    CDF sampling (`sample_cdf`) weighted by the same `imp_sample_dist`. This gives
201///    `sampled_excite` and its conditional probability `sampled_excite_prob`.
202///
203/// It then reconstructs the full `Excite` object, applies it to the source determinant's
204/// configuration, calculates the resulting determinant's coefficient (`H_ai * c_i`), and
205/// returns the source `Det`, the `Excite`, the resulting `Det` (with coefficient but `diag=None`),
206/// and the total probability (`det_orb_prob * sampled_excite_prob`).
207/// Returns `None` for the `Det` tuple if the sampled excitation was invalid (e.g., target occupied).
208///
209/// # Arguments
210/// * `screened_sampler`: The pre-computed sampler for source det/orb pairs.
211/// * `imp_sample_dist`: The importance sampling distribution to use for both steps.
212/// * `excite_gen`: Contains the lists of target excitations.
213/// * `ham`: The Hamiltonian operator (needed to compute the final coefficient).
214/// * `rand`: Mutable random number generator state.
215///
216/// # Returns
217/// A tuple `(Option<(source_det, excitation, target_det)>, total_probability)`.
218pub fn matmul_sample_remaining(
219    screened_sampler: &ScreenedSampler,
220    imp_sample_dist: ImpSampleDist,
221    excite_gen: &ExciteGenerator,
222    ham: &Ham,
223    rand: &mut Rand,
224) -> (Option<(Det, Excite, Det)>, f64) {
225    // Nothing to sample (everything was captured deterministically): report no
226    // valid sample rather than drawing from an empty distribution, which panics.
227    if screened_sampler.is_empty() {
228        return (None, 1.0);
229    }
230
231    let det_orb_sample: DetOrbSample;
232    let det_orb_prob: f64;
233    let sampled_excite: &StoredExcite;
234    let sampled_excite_prob: f64;
235
236    match imp_sample_dist {
237        ImpSampleDist::AbsHc => {
238            // First, sample a (determinant, orbs) pair using Alias sampling with prob |Hc|
239            let sample = screened_sampler
240                .det_orb_sampler_abs_hc
241                .sample_with_prob(rand);
242            det_orb_sample = screened_sampler.elements[sample.0];
243            det_orb_prob = sample.1;
244            // println!("Sampled det {}, prob {}", det_orb_sample, det_orb_prob);
245
246            // Sample excitation from this det/orb pair by binary search the stored cdf with prob |H|
247            match det_orb_sample.is_alpha {
248                None => {
249                    // Opposite spin double
250                    let sample2 = sample_cdf(
251                        excite_gen
252                            .opp_doub_sorted_list
253                            .get(&det_orb_sample.init)
254                            .unwrap(),
255                        &ImpSampleDist::AbsHc,
256                        Some(det_orb_sample.sum_abs_h),
257                        rand,
258                    )
259                    .unwrap();
260                    sampled_excite = sample2.0;
261                    sampled_excite_prob = sample2.1;
262                }
263                Some(_) => {
264                    match det_orb_sample.init {
265                        Orbs::Double(_) => {
266                            // Same spin double
267                            let sample2 = sample_cdf(
268                                excite_gen
269                                    .same_doub_sorted_list
270                                    .get(&det_orb_sample.init)
271                                    .unwrap(),
272                                &ImpSampleDist::AbsHc,
273                                Some(det_orb_sample.sum_abs_h),
274                                rand,
275                            )
276                            .unwrap();
277                            sampled_excite = sample2.0;
278                            sampled_excite_prob = sample2.1;
279                        }
280                        Orbs::Single(_) => {
281                            // Single
282                            let sample2 = sample_cdf(
283                                excite_gen
284                                    .sing_sorted_list
285                                    .get(&det_orb_sample.init)
286                                    .unwrap(),
287                                &ImpSampleDist::AbsHc,
288                                Some(det_orb_sample.sum_abs_h),
289                                rand,
290                            )
291                            .unwrap();
292                            sampled_excite = sample2.0;
293                            sampled_excite_prob = sample2.1;
294                        }
295                    }
296                }
297            }
298        }
299        ImpSampleDist::HcSquared => {
300            // First, sample a (determinant, orbs) pair using Alias sampling with prob (Hc)^2
301            let sample = screened_sampler
302                .det_orb_sampler_hc_squared
303                .sample_with_prob(rand);
304            det_orb_sample = screened_sampler.elements[sample.0];
305            det_orb_prob = sample.1;
306
307            // Sample excitation from this det/orb pair by binary search the stored cdf with prob H^2
308            match det_orb_sample.is_alpha {
309                None => {
310                    // Opposite spin double
311                    let sample2 = sample_cdf(
312                        excite_gen
313                            .opp_doub_sorted_list
314                            .get(&det_orb_sample.init)
315                            .unwrap(),
316                        &ImpSampleDist::HcSquared,
317                        Some(det_orb_sample.sum_h_squared),
318                        rand,
319                    )
320                    .unwrap();
321                    sampled_excite = sample2.0;
322                    sampled_excite_prob = sample2.1;
323                }
324                Some(_) => {
325                    match det_orb_sample.init {
326                        Orbs::Double(_) => {
327                            // Same spin double
328                            let sample2 = sample_cdf(
329                                excite_gen
330                                    .same_doub_sorted_list
331                                    .get(&det_orb_sample.init)
332                                    .unwrap(),
333                                &ImpSampleDist::HcSquared,
334                                Some(det_orb_sample.sum_h_squared),
335                                rand,
336                            )
337                            .unwrap();
338                            sampled_excite = sample2.0;
339                            sampled_excite_prob = sample2.1;
340                        }
341                        Orbs::Single(_) => {
342                            // Single
343                            let sample2 = sample_cdf(
344                                excite_gen
345                                    .sing_sorted_list
346                                    .get(&det_orb_sample.init)
347                                    .unwrap(),
348                                &ImpSampleDist::HcSquared,
349                                Some(det_orb_sample.sum_h_squared),
350                                rand,
351                            )
352                            .unwrap();
353                            sampled_excite = sample2.0;
354                            sampled_excite_prob = sample2.1;
355                        }
356                    }
357                }
358            }
359        }
360    }
361
362    // Construct the excitation (for output)
363    let excite = Excite {
364        init: det_orb_sample.init,
365        target: sampled_excite.target,
366        abs_h: sampled_excite.abs_h,
367        is_alpha: det_orb_sample.is_alpha,
368    };
369
370    // Apply the excitation to the sampled det
371    let sampled_det = det_orb_sample.det.config.safe_excite_det(&excite);
372
373    // Compute total probability
374    let prob_sampled_det: f64 = det_orb_prob * sampled_excite_prob;
375
376    match sampled_det {
377        None => (None, prob_sampled_det), // Proposed excitation would excite to already-occupied orbs
378        Some(d) => {
379            // Compute new det coefficient
380            let mut new_det_coeff = det_orb_sample.det.coeff;
381            match det_orb_sample.init {
382                Orbs::Double(_) => new_det_coeff *= ham.ham_doub(&det_orb_sample.det.config, &d),
383                Orbs::Single(_) => new_det_coeff *= ham.ham_sing(&det_orb_sample.det.config, &d),
384            }
385            (
386                Some((
387                    *det_orb_sample.det,
388                    excite,
389                    Det {
390                        config: d,
391                        coeff: new_det_coeff,
392                        diag: None, // Compute diagonal element later, only if needed (since it would be the most expensive step)
393                    },
394                )),
395                prob_sampled_det,
396            )
397        }
398    }
