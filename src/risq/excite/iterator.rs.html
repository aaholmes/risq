1//! # Excitation Iterators (`excite::iterator`)
2//!
3//! This module provides convenient iterators for generating excitations based on the
4//! pre-computed data in `ExciteGenerator`. These iterators handle the logic of
5//! iterating through determinants, initial orbitals, and target orbitals, applying
6//! screening thresholds and validity checks.

15/// Creates an iterator yielding deterministically significant external excitations.
16///
17/// Iterates through each determinant (`det`) in the input wavefunction `wf`.
18/// For each `det`, it iterates through all possible single and double excitations
19/// originating from its occupied valence orbitals, using the pre-sorted lists in `excite_gen`.
20///
21/// It yields tuples `(&det, excite, excited_config)` only for excitations that satisfy:
22/// 1. The estimated contribution `|H_ij * c_i|` is greater than or equal to `eps`.
23///    (Note: The `take_while` implies screening based on `|H_ij| * |c_i| >= eps`, assuming `abs_h` is `|H_ij|`).
24/// 2. The excitation is valid (target orbitals are unoccupied in `det`).
25/// 3. The resulting `excited_config` is *not* already present in the input `wf` (i.e., it's external).
26///
27/// The `excite` object in the output tuple is a full `Excite` struct, reconstructed from the `StoredExcite`.
28///
29/// # Arguments
30/// * `wf`: The source wavefunction.
31/// * `excite_gen`: The pre-computed excitation generator.
32/// * `eps`: The screening threshold for deterministic treatment.
33pub fn dets_excites_and_excited_dets<'a>(
34    wf: &'a Wf,
35    excite_gen: &'a ExciteGenerator,
36    eps: f64,
37) -> impl Iterator<Item = (&'a Det, Excite, Config)> {
38    wf.dets
39        .iter() // For each det
40        .flat_map(move |det| repeat(det).zip(valence_elecs_and_epairs(&det.config, excite_gen))) // For each electron and electron pair (det, (is_alpha, init_orbs))
41        .flat_map(move |(det, (is_alpha, init))| {
42            repeat((det, is_alpha, init))
43                .zip(excite_gen.excites_from((is_alpha, &init))) // For each (det, excite)
44                .take_while(move |((det, _is_alpha, _init), excite)| {
45                    excite.abs_h * det.coeff.abs() >= eps
46                }) // Stop searching for excites if eps threshold reached
47        })
48        .filter(move |((det, is_alpha, _), excite)| det.config.is_valid_stored(is_alpha, excite)) // Filter excites to already-occupied orbitals
49        .map(move |((det, is_alpha, init), excite)| {
50            (
51                det,
52                is_alpha,
53                init,
54                excite,
55                det.config.apply_excite(is_alpha, &init, excite),
56            )
57        }) // Compute excited determinant configuration
58        .filter(move |(_det, _is_alpha, _init, _excite, excited_det)| {
59            !wf.inds.contains_key(&excited_det)
60        }) // Filter excites to determinants in the exciting wavefunction
61        .map(move |(det, is_alpha, init, excite, excited_det)| {
62            (
63                det,
64                Excite {
65                    is_alpha,
66                    init,
67                    target: excite.target,
68                    abs_h: excite.abs_h,
69                },
70                excited_det,
71            ) // Convert StoredExcite to Excite
72        })
73}

77/// Creates an iterator yielding tuples of (det_index, det, spin, initial_orbs).
78///
79/// This iterates through each determinant `det` in the wavefunction `wf` and, for each `det`,
80/// yields all possible single and double *initial* orbital combinations (`init`) from its
81/// occupied valence orbitals, along with the corresponding spin channel (`is_alpha`).
82///
83/// Useful as a starting point for generating all excitations *from* the wavefunction.
84///
85/// # Arguments
86/// * `wf`: The source wavefunction.
87/// * `excite_gen`: Used only to access the `valence` orbital mask.
88pub fn dets_and_excitable_orbs<'a>(
89    wf: &'a Wf,
90    excite_gen: &'a ExciteGenerator,
91) -> impl Iterator<Item = (usize, &'a Det, Option<bool>, Orbs)> {
92    wf.dets
93        .iter()
94        .enumerate() // For each det
95        .flat_map(move |(ind, det)| {
96            repeat((ind, det)).zip(valence_elecs_and_epairs(&det.config, excite_gen))
97        }) // For each electron and electron pair (det, (is_alpha, init_orbs))
98        .map(move |((ind, det), (is_alpha, init))| (ind, det, is_alpha, init))
99}

101/// Creates an iterator yielding valid, external excitations from a specific determinant and initial orbital set.
102///
103/// Given a source determinant `det`, a spin channel `is_alpha`, and initial orbitals `init`,
104/// this iterates through all potential target excitations (`StoredExcite`) stored in `excite_gen`.
105///
106/// It yields tuples `(&stored_excite, excited_config)` only for excitations that are:
107/// 1. Valid (target orbitals are unoccupied in `det`).
108/// 2. External (the resulting `excited_config` is not already in `wf`).
109///
110/// Note: This iterator does *not* apply the `eps` screening threshold.
111///
112/// # Arguments
113/// * `det`: The source determinant.
114/// * `is_alpha`: The spin channel of the excitation.
115/// * `init`: The initial orbitals the excitation originates from.
116/// * `wf`: The wavefunction (used to check if the target is external).
117/// * `excite_gen`: The pre-computed excitation generator.
118pub fn excites_from_det_and_orbs<'a>(
119    det: &'a Det,
120    is_alpha: Option<bool>,
121    init: Orbs,
122    wf: &'a Wf,
123    excite_gen: &'a ExciteGenerator,
124) -> impl Iterator<Item = (&'a StoredExcite, Config)> {
125    repeat((det, is_alpha, init))
126        .zip(excite_gen.excites_from((is_alpha, &init))) // For each (det, excite)
127        .filter(move |((det, is_alpha, _), excite)| det.config.is_valid_stored(is_alpha, excite)) // Filter excites to already-occupied orbitals
128        .map(move |((det, is_alpha, init), excite)| {
129            (excite, det.config.apply_excite(is_alpha, &init, excite))
130        }) // Compute excited determinant configuration
131        .filter(move |(_excite, excited_det)| !wf.inds.contains_key(&excited_det))
132    // Filter excites to determinants in the exciting wavefunction
133}

137/// Creates an iterator yielding deterministically significant excitations *from* a single determinant.
138///
139/// Given a source determinant `det`, iterates through all possible single and double
140/// excitations originating from its occupied valence orbitals.
141///
142/// Yields tuples `(is_alpha, init_orbs, &stored_excite)` for excitations that satisfy:
143/// 1. The estimated contribution `|H_ij * c_i|` is greater than or equal to `eps`.
144/// 2. The excitation is valid (target orbitals are unoccupied in `det`).
145///
146/// This is useful for the HCI "search" step where new determinants are added based on `eps`.
147///
148/// # Arguments
149/// * `det`: The source determinant.
150/// * `excite_gen`: The pre-computed excitation generator.
151/// * `eps`: The screening threshold.
152pub fn excites<'a>(
153    det: &'a Det,
154    excite_gen: &'a ExciteGenerator,
155    eps: f64,
156) -> impl Iterator<Item = (Option<bool>, Orbs, &'a StoredExcite)> {
157    valence_elecs_and_epairs(&det.config, excite_gen) // For each electron and electron pair (is_alpha, init_orbs)
158        .flat_map(move |(is_alpha, init)| {
159            repeat((is_alpha, init))
160                .zip(excite_gen.excites_from((is_alpha, &init))) // For each (det, excite)
161                .take_while(move |((_is_alpha, _init), excite)| {
162                    excite.abs_h * det.coeff.abs() >= eps
163                })
164            // Stop searching for excites if eps threshold reached
165        })
166        .filter(move |((is_alpha, _), excite)| det.config.is_valid_stored(is_alpha, excite)) // Filter excites to already-occupied orbitals
167        .map(move |((is_alpha, init), excite)| (is_alpha, init, excite))
168}

170impl ExciteGenerator {
171    /// Helper function to get an iterator over `StoredExcite` for a given initial orbital set and spin.
172    ///
173    /// Takes `init = (is_alpha, initial_orbs)` and returns an iterator over the corresponding
174    /// pre-sorted `Vec<StoredExcite>` stored in the `ExciteGenerator`'s HashMaps.
175    /// Panics if the `initial_orbs` key is not found in the corresponding map.
176    fn excites_from(&self, init: (Option<bool>, &Orbs)) -> impl Iterator<Item = &StoredExcite> {
177        match init.1 {
178            Orbs::Double(_) => {
179                match init.0 {
180                    None => {
181                        // Opposite-spin double
182                        self.opp_doub_sorted_list.get(init.1).unwrap().iter()
183                    }
184                    Some(_) => {
185                        // Same-spin double
186                        self.same_doub_sorted_list.get(init.1).unwrap().iter()
187                    }
188                }
189            }
190            Orbs::Single(_) => {
191                // Single
192                self.sing_sorted_list.get(init.1).unwrap().iter()
193            }
194        }
195    }
196}
