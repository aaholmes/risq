1//! # Epstein-Nesbet Perturbation Theory Module (`pt`)
2//!
3//! This module implements the second-order Epstein-Nesbet perturbation theory (ENPT2)
4//! correction, often used in conjunction with Heat-bath Configuration Interaction (HCI)
5//! to form the Semistochastic HCI (SHCI) method.
6//!
7//! It includes functions for both purely deterministic PT (using a screening threshold `eps`)
8//! and semistochastic approaches that combine deterministic treatment of important
9//! contributions with stochastic sampling of the remaining perturbative space.

24/// Orchestrates the calculation of the second-order Epstein-Nesbet perturbation theory correction (PT2).
25///
26/// Based on the `global.use_new_semistoch` flag, it dispatches to either the older
27/// semistochastic method (`old_semistoch_enpt2`) or a newer importance-sampled variant
28/// (`new_semistoch_enpt2_dtm_diag_singles`). It then prints the final variational energy,
29/// PT2 correction (with standard deviation if sampled), and total energy.
30///
31/// # Arguments
32/// * `global`: Global calculation parameters.
33/// * `ham`: The Hamiltonian operator.
34/// * `excite_gen`: Pre-computed excitation generator data.
35/// * `wf`: The converged variational wavefunction (from HCI).
36pub fn perturbative(global: &Global, ham: &Ham, excite_gen: &ExciteGenerator, wf: &Wf) {
37    // Initialize random number genrator
38    let mut rand: Rand = init_rand().unwrap(); // TODO: Pass this in context
39
40    let e_pt2: f64;
41    let std_dev: f64;
42    // e_pt2 = dtm_pt(wf, excite_gen, ham, 1e-6);
43    // println!("Variational energy: {}, Deterministic PT: {}, Total energy: {}", wf.energy, e_pt2, wf.energy + e_pt2);
44    if !global.use_new_semistoch {
45        // Old SHCI (2017 paper)
46        println!("\nCalling semistoch ENPT2 the old way with p ~ |c|");
47        let out = old_semistoch_enpt2(wf, global, ham, excite_gen, false, &mut rand);
48        e_pt2 = out.0;
49        std_dev = out.1;
50    } else {
51        println!("\nCalling semistoch ENPT2 the new way with importance sampling");
52        let out = new_semistoch_enpt2_dtm_diag_singles(wf, global, ham, excite_gen, &mut rand);
53        e_pt2 = out.0;
54        std_dev = out.1;
55    }
56    println!("Variational energy: {:.6}", wf.energy);
57    println!("PT energy: {:.6} +- {:.6}", e_pt2, std_dev);
58    println!("Total energy: {:.6} +- {:.6}", wf.energy + e_pt2, std_dev);
59}

61/// Calculates the deterministic second-order Epstein-Nesbet PT correction.
62///
63/// 1. Generates the first-order interacting wavefunction `|psi_1> = sum_a |a><a|H|psi_0> / (E_0 - E_a)`,
64///    where `|psi_0>` is the input variational wavefunction `wf`, and `|a>` are external determinants.
65///    It uses `dets_excites_and_excited_dets` to find external determinants `|a>` connected
66///    to `|psi_0>` with `|H_ai * c_i| >= eps`.
67/// 2. Efficiently computes the diagonal elements `E_a = <a|H|a>` for these external determinants.
68/// 3. Calculates the PT2 energy using the formula: `E_PT2 = sum_a |<a|H|psi_0>|^2 / (E_0 - E_a)`.
69///
70/// # Arguments
71/// * `wf`: The converged variational wavefunction (`|psi_0>`).
72/// * `excite_gen`: Pre-computed excitation generator.
73/// * `ham`: The Hamiltonian operator.
74/// * `eps`: The screening threshold for including perturbative contributions.
75///
76/// # Returns
77/// The deterministic PT2 energy correction.
78pub fn dtm_pt(wf: &Wf, excite_gen: &ExciteGenerator, ham: &Ham, eps: f64) -> f64 {
79    println!("Start of deterministic PT");
80    let mut h_psi: Wf = Wf::default();
81    let mut n: usize = 0;
82    let mut old_var_det: Config = wf.dets[0].config;
83    for (var_det, excite, pt_config) in dets_excites_and_excited_dets(wf, excite_gen, eps) {
84        if var_det.config != old_var_det {
85            n += 1;
86            old_var_det = var_det.config;
87            if n % 1000 == 0 {
88                println!("Var det {} of {}", n, wf.n);
89            }
90        }
91        // println!("Var det: {}, PT det: {}", var_det, pt_config);
92
93        // Compute off-diagonal element times var_det.coeff
94        let h_ai_c_i: f64 = ham.ham_off_diag(&var_det.config, &pt_config, &excite) * var_det.coeff;
95
96        // For single excitaitons: Check whether this excite actually exceeds eps
97        if let Orbs::Single(_) = excite.init {
98            if h_ai_c_i.abs() < eps {
99                continue;
100            }
101        }
102
103        // Compute diagonal element in O(N) time only if necessary
104        if let Some(ind) = h_psi.inds.get_mut(&pt_config) {
105            h_psi.dets[*ind].coeff += h_ai_c_i;
106        } else {
107            h_psi.dets.push(Det {
108                config: pt_config,
109                coeff: h_ai_c_i,
110                diag: Some(var_det.new_diag(ham, &excite)),
111            });
112        }
113    }
114    println!("Preparing to calculate PT energy from generated PT dets");
115    pt(&h_psi, wf.energy)
116}

120/// Evaluate the PT expression given H * Psi and E0
121/// Calculates the PT2 energy sum given the first-order interacting wavefunction `h_psi`.
122///
123/// Computes `sum_a |(H*psi)_a|^2 / (E0 - E_a)`, where `a` iterates over the determinants
124/// in `h_psi`, `(H*psi)_a` is `det.coeff`, `E_a` is `det.diag`, and `E0` is the reference energy.
125/// Assumes `det.diag` is `Some` for all determinants in `h_psi`.
126pub fn pt(h_psi: &Wf, e0: f64) -> f64 {
127    h_psi.dets.iter().fold(0.0f64, |e_pt, det| {
128        e_pt + det.coeff * det.coeff / (e0 - det.diag.unwrap())
129    })
130}

132/// Evaluate off-diagonal component of the PT expression given H * Psi, sum_sq, and E0
133/// Calculates the off-diagonal component of a PT2 estimator (used in stochastic methods).
134///
135/// Computes `sum_a [ ((H*psi)_a)^2 - (sum_sq)_a ] / (E0 - E_a)`.
136/// This form often appears in variance calculations or specific stochastic PT estimators
137/// where `sum_sq` might represent the sum of squared individual contributions `(H_ai * c_i)^2`.
138/// Assumes `h_psi` and `sum_sq` have corresponding determinants with valid diagonal energies.
139pub fn pt_off_diag(h_psi: &Wf, sum_sq: &Wf, e0: f64) -> f64 {
140    // h_psi.dets.iter().zip(sum_sq.dets.iter()).fold(0.0f64, |e_pt, (sum_hc, sum_hc_sq)| {
141    //     e_pt + (sum_hc.coeff * sum_hc.coeff - sum_hc_sq.coeff) / (e0 - sum_hc.diag.unwrap())
142    // })
143    let mut out: f64 = 0.0;
144    for (sum_hc, sum_hc_sq) in h_psi.dets.iter().zip(sum_sq.dets.iter()) {
145        out += (sum_hc.coeff * sum_hc.coeff - sum_hc_sq.coeff) / (e0 - sum_hc.diag.unwrap());
146    }
147    out
148}

150/// Stores samples collected during stochastic or semistochastic PT calculations.
151///
152/// Used to accumulate information needed to compute the unbiased PT2 energy estimator
153/// as described in the SHCI papers.
154///
155/// The main `samples` HashMap maps a perturbative determinant configuration (`Config`)
156/// to a tuple containing:
157/// 1.  `f64`: The diagonal energy (E_a) of the perturbative determinant `a`.
158/// 2.  `HashMap<Config, (f64, f64, i32)>`: A nested map where keys are the configurations
159///     of *variational* determinants (`i`) that generated this perturbative determinant `a`.
160///     The values are tuples `(H_ai * c_i, p_ai, w_ai)`:
161///     *   `H_ai * c_i`: The matrix element connecting `a` and `i` times the coefficient of `i`.
162///     *   `p_ai`: The probability with which the excitation `i -> a` was sampled.
163///     *   `w_ai`: The number of times this specific excitation `i -> a` was sampled (weight).
164#[derive(Default)]
165pub struct PtSamples {
166    /// Total number of samples collected.
167    pub n: i32,
168    /// The main storage for collected samples.
169    pub samples: HashMap<Config, (f64, HashMap<Config, (f64, f64, i32)>)>,
170}

191    /// Adds a new sample to the collection, computing the PT determinant's diagonal if needed.
192    ///
193    /// Takes a sampled excitation from `var_det` to `pt_det` (via `excite`) with probability
194    /// `sampled_prob`. It updates the nested HashMaps in `self.samples`. If `pt_det.config`
195    /// is encountered for the first time, its diagonal energy `E_a` is computed efficiently
196    /// using `var_det.new_diag` and stored.
197    pub fn add_sample_compute_diag(
198        &mut self,
199        var_det: Det,        // The variational determinant (i) that was sampled from
200        excite: &Excite,     // The excitation (i -> a) that was sampled
201        pt_det: Det,         // The resulting perturbative determinant (a), contains H_ai * c_i in coeff field
202        sampled_prob: f64,   // The probability p_ai with which this excitation was sampled
203        ham: &Ham,           // Hamiltonian needed for new_diag
204    ) {
205        // Add a new sample to PtSamples
206        // Compute diagonal element of perturbative determinant if it hasn't already been computed
207        self.n += 1;
208        // match excite.init {
209        //     Orbs::Double(_) => {
210        //         println!("Doubles in add_sample_compute_diag: (Hc)^2 / p = {}", pt_det.coeff * pt_det.coeff / sampled_prob);
211        //     },
212        //     Orbs::Single(_) => {
213        //         println!("Singles in add_sample_compute_diag: (Hc)^2 / p = {}", pt_det.coeff * pt_det.coeff / sampled_prob);
214        //     }
215        // }
216        match self.samples.get_mut(&pt_det.config) {
217            None => {
218                // New PT det was sampled: compute diagonal element and create new variational det map
219                let pt_diag_elem: f64 = var_det.new_diag(ham, excite);
220                let mut var_det_map: HashMap<Config, (f64, f64, i32)> = Default::default();
221                var_det_map.insert(var_det.config, (pt_det.coeff, sampled_prob, 1));
222                self.samples
223                    .insert(pt_det.config, (pt_diag_elem, var_det_map));
224            }
225            Some(pt_det_info) => {
226                // No need to recompute the diagonal element
227                Self::process_resampled_pt_det(&var_det, pt_det, sampled_prob, pt_det_info)
228            }
229        }
230    }

232    /// Helper function to update the sample information when a perturbative determinant `pt_det`
233    /// has already been encountered (i.e., its diagonal energy is known).
234    ///
235    /// Updates the inner HashMap associated with `pt_det.config`. If the specific variational
236    /// determinant `var_det` is new for this `pt_det`, it's added. If `var_det` has contributed
237    /// to this `pt_det` before, only its weight (`w_ai`) is incremented.
238    fn process_resampled_pt_det(
239        var_det: &Det,          // The variational determinant (i)
240        pt_det: Det,            // The perturbative determinant (a)
241        sampled_prob: f64,      // The sampling probability p_ai
242        pt_det_info: &mut (f64, HashMap<Config, (f64, f64, i32)>), // Existing info for pt_det
243    ) {
244        match pt_det_info.1.get_mut(&var_det.config) {
245            None => {
246                // New var det to reach this PT det; add to variational det map
247                pt_det_info
248                    .1
249                    .insert(var_det.config, (pt_det.coeff, sampled_prob, 1));
250            }
251            Some(var_det_info) => {
252                // Already have this var det; just increment number of times it has been sampled
253                var_det_info.2 += 1;
254            }
255        }
256    }

258    /// Adds a new sample, assuming the PT determinant's diagonal energy is already stored.
259    ///
260    /// Similar to `add_sample_compute_diag`, but assumes `pt_det.diag` contains the
261    /// correct diagonal energy `E_a`, avoiding the call to `new_diag`.
262    pub fn add_sample_diag_already_stored(&mut self, var_det: Det, pt_det: Det, sampled_prob: f64) {
263        // Add a new sample to PtSamples
264        // Assumes that pt_det's diagonal element already stored
265        self.n += 1;
266        match self.samples.get_mut(&pt_det.config) {
267            None => {
268                // New PT det was sampled
269                let mut var_det_map: HashMap<Config, (f64, f64, i32)> = Default::default();
270                var_det_map.insert(var_det.config, (pt_det.coeff, sampled_prob, 1));
271                self.samples
272                    .insert(pt_det.config, (pt_det.diag.unwrap(), var_det_map));
273            }
274            Some(pt_det_info) => {
275                // No need to recompute the diagonal element
276                Self::process_resampled_pt_det(&var_det, pt_det, sampled_prob, pt_det_info)
277            }
278        }
279    }

287    /// where N_v (`n_det`) is the number of variational determinants sampled from in total.
288    ///
